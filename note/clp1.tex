\section{Numbers}\label{sec_0_1}
\begin{itemize}
\item Natural numbers $\mathbb{N}$
\begin{itemize}
\item $1$, $2$, $3$, $\dots$
\item No $0$ !
\item $\mathbb{N}$ is \emph{closed} under addition and multiplication
\item Prime numbers
\begin{itemize}
\item When the only natural numbers that divide it exactly are 1 and itself
\item Unique factorization of integer
\item 1 is not prime: otherwise $1\times2\times 3$ and $2\times 3$ are different factorizations
\end{itemize}
\item Composite numbers --- a natural number is composite when it is not prime
\item Ex. $7$ is prime, but $6 = 3\times 2$ is composite
\end{itemize}
\item Integers $\mathbb{Z}$
\begin{itemize}
\item all positive and negative numbers together with the number zero
\item $\mathbb{Z}$ is closed under addition, subtraction and multiplication, but not division
\item
\begin{itemize}
\item Even numbers %--- an integer is even if it is exactly divisible by $2$, or equivalently if it can be written as the product of 2 and another integer. This means that $-14, 6$ and $0$ are all even.
\item Odd numbers %--- an integer is odd when it is not even. Equivalently it can be written as $2k+1$ where $k$ is another integer. Thus $11 = 2\times 5+1$ and $-7 = 2\times(-4)+1$ are both odd.
\end{itemize}
\end{itemize}
\item Rational numbers $\mathbb{Q}$
\begin{itemize}
\item Any rational number $r$ can be written as $\frac{p}{q}$ where $p,q$ are integers
\item $\mathbb{Q}$ is closed under addition, subtraction, multiplication and division
\end{itemize}
\item Real numbers $\mathbb{R}$
\begin{itemize}
\item Numbers that can be written as decimal expansions
\item $\sqrt{2}$, $\pi$, $e$ (the most important constant in calculus)
\item Fact: between any two real numbers one can always find a rational number
\item $\mathbb{Q}$ is \emph{dense} in $\mathbb{R}$
\end{itemize}
\end{itemize}
\begin{defn}[Sets of numbers]\label{def_0_1_1}
\begin{itemize}
\item $\mathbb{N} = $ the natural numbers
\item $\mathbb{Z} = $ the integers
\item $\mathbb{Q} = $ the rationals
\item $\mathbb{R} = $ the reals
\end{itemize}
\end{defn}

\begin{theorem}\label{thm_0_1_1}
Let $x\in\mathbb{R}$. Then either % $x$ must fall into one of the following two categories:
\begin{itemize}
\item $x$ has a unique decimal expansion
\item $x$ is a rational number of the form $\frac{a}{2^k 5^\ell}$ where $a\in\mathbb{Z}$ and $k,l$ are non-negative integers.

\section{Sets}\label{sec sets}
\begin{defn}[A not-so-formal definition of set]\label{def_0_2_1}
A ``set'' is a collection of distinct objects. The objects are referred to as ``elements'' or ``members'' of the set.
\end{defn}
\begin{itemize}
\item Capital letters $A, B, C, X, Y$ denote sets
\item Lowercase letters $a, b, c, x, y$ denote set elements
\item $i, j, k, l, m, n$ denote $\mathbb{Z}$ (e.g. $1,2,3,-5,18,\dots$)
\item $x, y, z, w$ denote $\mathbb{R}$ (e.g. $1.4323, \pi, \sqrt{2}, 6.0221415\times 10^{23} \dots$)
\item
\begin{align*}
\text{``$4 \in A$''}
&&& \text{is read as} && \text{``Four is an element of  $A$.''} \\
\text{``$3 \notin A$''}
&&& \text{is read as} &&  \text{``Three is not an element of $A$.''}
\end{align*}
\item ``Let $C = \{1,2,3\}$.''. $C = \{1,2,3\} = \{2,1,3\} = \{3,2,1\}$
\end{itemize}

\end{align*}
\end{defn}

\begin{defn}\label{def function}
Let $A, B$ be non-empty sets.
\begin{itemize}
\item A function $f$ from $A$ to $B$, is a rule or formula that takes elements of $A$ as inputs and returns elements of $B$ as outputs.
\item We write this as
\begin{align*}
f: A \to B
\end{align*}
and if $f$ takes $a \in A$ as an input and returns $b\in B$ then we write this as $f(a) = b$.
\item Every function must satisfy the following two conditions:
\begin{itemize}
\item The function must be defined on every possible input from $A$.
\item The function is only allowed to return one result for each input.
\end{itemize}
\end{itemize}
\end{defn}
\begin{defn}\label{def_0_4_1}
Let $f:A \to B$ be a function. %Then
\begin{itemize}
\item $A$: the ``domain'' of $f$; $A=\text{dom}(f)$. %the set $A$ of inputs to our function is the ``domain'' of $f$,
\item $B$: the ``codomain'' of $f$; $B=\text{codom}(f)$. %the set $B$ which contains all the results is called the codomain,
\item $\text{range}(f) \subseteq B$ and is defined by
\begin{align*}
\text{range}(f)=\{b\in B\;|\; \exists a\in A\;(f(a) = b)\}
\end{align*}
\end{itemize}
\end{defn}
\begin{eg}[domains and ranges]\label{eg_0_4_1}
In ``days of the week'' function example
\begin{itemize}
\item The domain $A$ is the set of days of the week.
\item The codomain $B$ is the 26 letters of the alphabet.
\item The range is $\set{F,M,T,S,W}$ %--- no other elements of $B$ are images of inputs from $A$.
\end{itemize}
\end{eg}

\begin{defn}[Informal definition of one-sided limits]\label{def informal onesided limits}
\begin{itemize}
\item (left-hand limit, limit from below) We write
\begin{align*}
\lim_{x \to a^-} f(x) &= L
\end{align*}
when $f(x)$ gets closer and closer to $L$ whenever $x<a$ and $x$ moves closer and closer to $a$.
\item (right-hand limit, limit from above) We write
\begin{align*}
\lim_{x \to a^+} f(x) &= R
\end{align*}
when $f(x)$ gets closer and closer to $R$ whenever $x>a$ and $x$ moves closer and closer to $a$.
\end{itemize}
\end{defn}
\begin{align*}
\lim_{x \to a^+} f(x) &= \lim_{x \to a+} f(x) = \lim_{x \downarrow a} f(x) = \lim_{x \searrow a} f(x) = R & \text{right-hand limit}\\
\lim_{x \to a^-} f(x) &= \lim_{x \to a-} f(x) = \lim_{x \uparrow a} f(x) = \lim_{x \nearrow a} f(x) = L & \text{left-hand limit}
\end{align*}
\begin{theorem}[Limits and one sided limits]\label{thm_1_3_1}
\begin{align*}
\lim_{x \to a} f(x) = L && \mbox{ if and only if } && \lim_{x \to a^-} f(x) = L \mbox{ and } \lim_{x \to a^+} f(x) = L
\end{align*}
\end{theorem}

\begin{defn}\label{def lim is inf informal}
\begin{itemize}
\item We write
\begin{align*}
\lim_{x \to a} f(x) = + \infty
\end{align*}
when $f(x)$ becomes arbitrarily large and positive as $x$ gets closer and closer to $a$, without being exactly $a$.
\item We write
\begin{align*}
\lim_{x \to a} f(x) = - \infty
\end{align*}
when $f(x)$ becomes arbitrarily large and negative as $x$ gets closer and closer to $a$, without being exactly $a$.
\end{itemize}
\end{defn}
\begin{itemize}
\item function~1
\begin{align*}
\lim_{x\to a^-} f(x) &= +\infty & \lim_{x\to a^+} f(x) &= +\infty
\end{align*}
\item function~2
\begin{align*}
\lim_{x\to a^-} f(x) &= -\infty & \lim_{x\to a^+} f(x) &= -\infty
\end{align*}
\end{itemize}
\begin{align*}
\lim_{x \to 0} \frac{1}{x^2} &= +\infty & \lim_{x \to 0} -\frac{1}{x^2} &= -\infty
\end{align*}

\item We write
\begin{align*}
\lim_{x \to a^+} f(x) = + \infty
\end{align*}
when $f(x)$ becomes arbitrarily large and positive as $x$ gets closer and closer to $a$ from above, without being exactly $a$.
\item We write
\begin{align*}
\lim_{x \to a^+} f(x) = - \infty
\end{align*}
when $f(x)$ becomes arbitrarily large and negative as $x$ gets closer and closer to $a$ from above, without being exactly $a$.

\begin{theorem}[Easiest limits]\label{thm_easy_lim}
Let $a, c\in\mathbb{R}$, then
\begin{align*}
\lim_{x \to a} c & = c & \text{ and }&& \lim_{x \to a} x &= a.
\end{align*}
\end{theorem}
\begin{theorem}[Arithmetic of limits]\label{thm arith lim}
Let $a$, $c\in\mathbb{R}$, $f(x)$, $g(x)$ be defined for all $x$'s that lie in some interval about $a$ (but $f,g$ need not be defined exactly at $a$), and
\begin{align*}
\lim_{x \to a} f(x)&=F & \lim_{x \to a} g(x) &=G
\end{align*}
exist with $F,G \in \mathbb{R}$. Then
\begin{itemize}
\item $\ds \lim_{x \to a} (f(x) + g(x)) = F + G$ --- limit of the sum is the sum of the limits.
\item $\ds \lim_{x \to a} (f(x) - g(x)) = F - G$ --- limit of the difference is the difference of the limits.
\item $\ds \lim_{x \to a} c f(x) = c F$
\item $\ds \lim_{x \to a} ( f(x) \cdot g(x) ) = F \cdot G$ --- limit of the product is the product of limits.
\item If $G \neq 0$ then $\ds \lim_{x \to a} \frac{f(x)}{g(x)} = \frac{F}{G}$ and, in particular, $\ds \lim_{x\rightarrow a}\frac{1}{g(x)}=\frac{1}{G}$.
\end{itemize}
\end{theorem}

\begin{eg}[Be careful with limits of ratios]\label{eg lim rat}
\begin{align*}
\lim_{x\to a}f(x) = F, \quad\lim_{x\to a}g(x) = G = 0\quad\Longrightarrow\quad\lim_{x\to a}\frac{f(x)}{g(x)} = \text{undetermined!}

\begin{itemize}
\item If $F\not=0$:
\begin{itemize}
\item the limit does not exist, eg. $\ds \lim_{x\to 0} \frac{1}{x} = \text{DNE}$.
\item the limit is $\pm\infty$, eg. $\ds \lim_{x\to 0} \frac{\pm 1}{x^2} = +\infty$ or $\ds \lim_{x \to 0} \frac{-1}{x^2} = \lim_{x\to 0}\frac{-1}{x^2} = -\infty$.
\end{itemize}
\item If $F=0$:
\begin{itemize}
\item the limit does not exist, eg. $\ds \lim_{x \to 0} \frac{x}{x^2} = \lim_{x\to 0} \frac{1}{x} = \text{DNE}$
\item the limit is $\pm \infty$, eg. $\ds \lim_{x \to 0} \frac{x^2}{x^4} = \lim_{x\to 0} \frac{1}{x^2} = +\infty$ or $\ds \lim_{x \to 0} \frac{-x^2}{x^4} = \lim_{x\to 0} \frac{-1}{x^2} = -\infty$.
\item the limit is zero, eg. $\ds \lim_{x \to 0} \frac{x^2}{x} = 0$
\item the limit exists and is non-zero, eg. $\ds \lim_{x \to 0} \frac{x}{x} = 1$
\end{itemize}
\end{itemize}

\begin{theorem}[More arithmetic of limits --- powers and roots]\label{thm lim powers}
Let $n\in\mathbb{N}$, $a\in\mathbb{R}$, and function $f$ with
\begin{align*}
\lim_{x\to a} f(x) &= F
\end{align*}
for some $F\in\mathbb{R}$. Then %the following holds
\begin{align*}
\lim_{x\to a} \left(f(x) \right)^n = \left(\lim_{x\to a} f(x)\right)^n = F^n
\end{align*}
If
\begin{itemize}
\item $n$ is even and $F > 0$, or
\item $n$ is odd
\end{itemize}
then
\begin{align*}
\lim_{x \to a} \left( f(x) \right)^{1/n} = \left(\lim_{x \to a} f(x) \right)^{1/n} = F^{1/n}
\end{align*}
If $F > 0$ and $p\in\mathbb{R}$,
\begin{align*}
\lim_{x \to a} \left( f(x) \right)^p = \left(\lim_{x \to a} f(x) \right)^p = F^p
\end{align*}
\end{theorem}
\begin{itemize}
\item Notice that we have to be careful when taking roots of limits that might be negative numbers.
\item To see why, consider the case $n=2$, the limit
\begin{align*}
\lim_{x \to 4} x^{1/2} &= 4^{1/2} = 2 \\
\lim_{x \to 4} (-x)^{1/2} &= (-4)^{1/2} = \text{not a real number}
\end{align*}
\end{itemize}

\begin{defn}[Limits at infinity --- informal]\label{def_1_5_1}

\lim_{x \to -\infty} f(x) &= L
\end{align*}
when $f(x)$ gets closer and closer to $L$ as we make $x$ larger and larger and negative.
\end{itemize}

\begin{theorem}\label{thm basic lim inf}
Given $c\in\mathbb{R}$,
\begin{align*}
\lim_{x \to \infty} c &= c & \lim_{x \to -\infty} c &= c \\
\lim_{x \to \infty} \frac{1}{x} &= 0 & \lim_{x \to -\infty} \frac{1}{x} &= 0
\end{align*}
\end{theorem}
\begin{theorem}[Arithmetic of limits at infinity]\label{thm_1_5_1}
Given $f(x)$, $g(x)$ with
\begin{align*}
\lim_{x \to \infty} f(x)=F, \qquad\lim_{x \to \infty} g(x) = G,
\end{align*}
($F$, $G\in\mathbb{R}$). Then
\begin{align*}
\lim_{x \to \infty} f(x) \pm g(x) &= F \pm G \\
\lim_{x \to \infty} f(x) g(x) &= F  G \\
\lim_{x \to \infty} \frac{f(x)}{ g(x) } &= \frac{F}{G} & \text{provided $G \neq 0$}
\intertext{and for $p\in\mathbb{R}$}
\lim_{x \to \infty} f(x)^p &= F^p & \text{provided $F^p$ and $f(x)^p$ are defined for all $x$}
\end{align*}
The analogous results hold for limits to $-\infty$.
\end{theorem}
\begin{itemize}
\item $\forall r\in\mathbb{Q}$ with $r > 0$,
\begin{align*}
\lim_{x \to \infty} \frac{1}{x^r} &= 0
\end{align*}
\item Be careful with
\begin{align*}
\lim_{x \to -\infty} \frac{1}{x^r} &= 0
\end{align*}
This is true only if the least denomiator of $r$ is not even!
\item eg. $\ds \lim_{x \to \infty} \frac{1}{x^{1/2}} = 0$, but $\ds \lim_{x \to -\infty} \frac{1}{x^{1/2}}$ does not exist.%, because $x^{1/2}$ is not defined for $x<0$.
\item On the other hand, $x^{4/3}$ is defined for negative values of $x$ and $\ds \lim_{x \to -\infty} \frac{1}{x^{4/3}} = 0$.
\end{itemize}
\begin{eg}\label{eg_1_5_2}
Compute
\begin{align*}
\lim_{x \to \infty} \frac{x^2-3x+4}{3x^2+8x+1}

As $x$ becomes very large, $x^2$ term will dominate:
\begin{align*}
\frac{x^2-3x+4}{3x^2+8x+1}
&= \frac{x^2 \left(1-\frac{3}{x}+\frac{4}{x^2}\right)}
{x^2 \left(3+\frac{8}{x}+\frac{1}{x^2} \right)}\\
&= \frac{1-\frac{3}{x}+\frac{4}{x^2}}
{3+\frac{8}{x}+\frac{1}{x^2}} & \text{ remove the common factors}
\end{align*}
so
\begin{align*}
\lim_{x \to \infty} \frac{x^2-3x+4}{3x^2+8x+1}
&= \lim_{x \to \infty} \frac{1-\frac{3}{x}+\frac{4}{x^2}}
{3+\frac{8}{x}+\frac{1}{x^2}}\\
&= \frac{\ds \lim_{x \to \infty}\left(1-\frac{3}{x}+\frac{4}{x^2}\right)}
{\ds \lim_{x \to \infty}\left(3+\frac{8}{x}+\frac{1}{x^2} \right)}
& \text{arithmetic of limits} \\
&= \frac{\ds \lim_{x \to \infty} 1
- \lim_{x \to \infty} \frac{3}{x} + \lim_{x \to \infty} \frac{4}{x^2}}
{\ds \lim_{x \to \infty} 3

& \text{more arithmetic of limits} \\
&= \frac{1+0+0}{3+0+0}  = \frac{1}{3}
\end{align*}

\begin{theorem}[Arithmetic of infinite limits]
\label{thm arith inf lim}
Let $a,c,H \in \mathbb{R}$ and let $f,g,h$ be functions defined in
an interval around $a$ (but they need not be defined at $x=a$), so that
\begin{align*}
\lim_{x \to a} f(x) &= +\infty &
\lim_{x \to a} g(x) &= +\infty &
\lim_{x \to a} h(x) &= H
\end{align*}
Then
\begin{itemize}
\item $\ds \lim_{x \to a} ( f(x) + g(x) ) = +\infty$
\item $\ds \lim_{x \to a} ( f(x) + h(x) ) = +\infty$
\item $\ds \lim_{x \to a} ( f(x) - g(x) )$ undetermined
\item $\ds \lim_{x \to a} ( f(x) - h(x) ) = +\infty$
\item $\ds \lim_{x \to a} c f(x) =
\begin{cases}
+\infty & c>0 \\
-\infty & c<0 \\
0       & c=0
\end{cases}$
\item $\ds \lim_{x \to a} ( f(x) \cdot g(x) ) = +\infty$.
\item $\ds \lim_{x \to a} f(x) h(x) =
\begin{cases}
+\infty & H>0 \\
-\infty & H<0\\
\text{undetermined} & H=0
\end{cases}
$
\item $\ds \lim_{x \to a} \frac{f(x)}{g(x)}$ undetermined
\item $\ds \lim_{x \to a} \frac{f(x)}{h(x)} =
\begin{cases}
+\infty & H>0 \\
-\infty & H<0\\
\text{undetermined} & H=0
\end{cases}$
\item $\ds \lim_{x \to a} \frac{h(x)}{f(x)} = 0$
\item $\ds \lim_{x \to a} f(x)^p =
\begin{cases}
+\infty & p>0 \\
0 & p<0\\
1 & p=0
\end{cases}$
\end{itemize}
\end{theorem}
\begin{itemize}
\item Examples of
\begin{align*}
\lim_{x \to 0} f(x) = +\infty, \quad \lim_{x \to 0} h(x) = 0\quad\Longrightarrow\quad\lim_{x \to 0} f(x) h(x) = \text{undetermined}
\end{align*}
Take $f(x) = \frac{1}{x^4}$, $h(x) = x^2,\,x^3,\,x^4,\,x^5$, then
\begin{align*}
\lim_{x \to 0} f(x) h(x) =
\begin{cases}
\displaystyle\lim_{x \to 0} \frac{1}{x^4} x^2 = \lim_{x \to 0} \frac{1}{x^2} = +\infty \\
\displaystyle\lim_{x \to 0} \frac{1}{x^4} x^3 = \lim_{x \to 0} \frac{1}{x} = \text{DNE}\\
\displaystyle\lim_{x \to 0} \frac{1}{x^4} x^4 = \lim_{x \to 0} 1 = 1                   \\
\displaystyle\lim_{x \to 0} \frac{1}{x^4} x^5 = \lim_{x \to 0} x = 0                   \\
\end{cases}
\end{align*}
\end{itemize}
\begin{itemize}
\item Examples of
\begin{align*}
\lim_{x \to 0} f(x) = +\infty, \quad \lim_{x \to 0} g(x) = +\infty\quad\Longrightarrow\quad\lim_{x \to 0} \frac{f(x)}{g(x)} = \text{undetermined}
\end{align*}
Take $f(x) = \frac{1}{x^4}$, $g(x) = \frac{1}{x^6},\,\frac{1}{x^4},\,\frac{1}{x^2}$, then
\begin{align*}
\lim_{x \to 0} \frac{f(x)}{g(x)} = \begin{cases}
\displaystyle\lim_{x \to 0}\frac{\frac{1}{x^4}}{\frac{1}{x^6}} = \lim_{x \to 0} x^2 = 0 \\
\displaystyle\lim_{x \to 0}\frac{\frac{1}{x^4}}{\frac{1}{x^4}} = \lim_{x \to 0} 1 = 1 \\
\displaystyle\lim_{x \to 0}\frac{\frac{1}{x^4}}{\frac{1}{x^2}} = \lim_{x \to 0} \frac{1}{x^2} = +\infty
\end{cases}
\end{align*}
\end{itemize}

\section{Continuity}\label{sec_1_6}
\begin{defn}\label{def_1_6_1}
A function $f(x)$ is continuous at $a$ if
\begin{align*}
\lim_{x \to a} f(x) &= f(a).
\end{align*}
\begin{itemize}
\item $f$ is not continuous / discontinuous at $a$
\item $f$ is continuous: $f$ is continuous at $a$ for all $a \in \mathbb{R}$
\item $f(x)$ is continuous on $(a,b)$: $f$ is continuous $\forall c$ where $a < c <b$
\end{itemize}
\end{defn}
$f$ is continuous at $a$ then
\begin{itemize}
\item $f(a)$ exists
\item $\ds \lim_{x \to a^-}$ exists and is equal to $f(a)$, and
\item $\ds \lim_{x \to a^+}$ exists and is equal to $f(a)$.
\end{itemize}
\begin{defn}\label{def_1_6_2}
$f(x)$ is continuous \emph{from the right} at $a$ if
\begin{align*}
\lim_{x\to a^+} f(x) &= f(a).
\end{align*}
$f(x)$ is continuous \emph{from the left} at $a$ if
\begin{align*}
\lim_{x\to a^-} f(x) &= f(a)
\end{align*}
\end{defn}
\begin{defn}\label{def_1_6_3}
$f(x)$ is continuous on $[a,b]$ when
\begin{itemize}
\item $f(x)$ is continuous on $(a,b)$
\item $f(x)$ is continuous from the right at $a$
\item $f(x)$ is continuous from the left at $b$
\end{itemize}
Note that the last two condition are equivalent to
\begin{align*}
\lim_{x\to a^+} f(x) &= f(a) & \text{ and } && \lim_{x\to b^-} f(x) &= f(b).
\end{align*}
\end{defn}

\begin{theorem}[Compositions and continuity]\label{thm comp cont}
If $f$ is continuous at $b$ and $\ds \lim_{x \to a} g(x) = b$ then $\ds \lim_{x\to a} f(g(x)) = f(b)$. I.e.
\begin{align*}
\lim_{x \to a} f\big(g(x)\big) &= f\big(\lim_{x \to a} g(x)\big)
\end{align*}
Hence if $g$ is continuous at $a$ and $f$ is continuous at $g(a)$ then %the composite function
$(f \circ g)(x) = f(g(x))$ is continuous at $a$.
\end{theorem}

\begin{defn}\label{def_1_7_1}
Let $a \in \mathbb{R}$ and %let
$f(x)$ be a function defined everywhere in a neighbourhood of $a$, except possibly at $a$. %We say that
Then
\begin{align*}
\lim_{x \to a} f(x) &= L
\end{align*}
if and only if for every $\epsilon >0$ there exists $\delta>0$ so that
\begin{align*}
\text{if } 0 < |x-a| < \delta \text{ then } |f(x) - L| <\epsilon.
\end{align*}
In symbols:
\begin{align*}
\lim_{x \to a} f(x) = L\,\,\Longleftrightarrow\,\,\forall\,\epsilon > 0\,\,\exists\,\delta > 0\,(0 < |x - a| < \delta\,\Longrightarrow\,|f(x) - L| < \epsilon)
\end{align*}
\end{defn}
\begin{itemize}
\item $\lim_{x \to a} f(x) = L$ via the $\epsilon$-$\delta$ argument: Given an arbitrary $\epsilon > 0$, we should be able to find $\delta > 0$ which makes $|f(x) - L|< \epsilon$ if $0 < |x - a| < \delta$.
\end{itemize}
\begin{itemize}
\item Using $\epsilon$-$\delta$ argument to prove
\begin{align*}
\lim_{x\to 1}x^2 = 1
\end{align*}
\item
Let $\epsilon > 0$ be given. If we can find $\delta$ which makes $|x^2 - 1| < \epsilon$ when $0 < |x - 1| < \delta$ then we're done.
\item Note that $|x^2 - 1| = |(x + 1)(x - 1)|$. If we assume $\delta\leqslant 1$, then from
\begin{align*}
|x - 1| < 1
\end{align*}
we have
\begin{gather*}
-1 < x - 1 < 1 \\
0 < x < 2 \\
1 < x + 1 < 3 \\
|x + 1| < 3
\end{gather*}
Now take $\delta = \min\{1, \frac{\epsilon}{3}\}$; when $0 < |x - 1| < \delta$,
\begin{align*}
|x^2 - 1| = |(x - 1)(x + 1)| < |x + 1||x - 1| < 3\delta \leqslant 3\cdot\frac{\epsilon}{3}= \epsilon,
\end{align*}
as required.
\end{itemize}
\begin{itemize}
\item Using $\epsilon$-$\delta$ argument to prove
\begin{align*}
\lim_{x\to 2}x^2 = 4
\end{align*}
\item

\begin{align*}
|x - 2| < 1

-1 < x - 2 < 1 \\
1 < x < 3 \\
3 < x + 2 < 5 \\
|x + 2| < 5
\end{gather*}

\end{align*}
as required.
\end{itemize}

\begin{align*}
\lim_{x\to 3}x^2 = 9
\end{align*}

Let $\epsilon > 0$ be given. If we can find $\delta$ which makes $|x^2 - 9| < \epsilon$ when $0 < |x - 3| < \delta$ then we're done.

|x - 3| < 1

-1 < x - 3 < 1 \\
2 < x < 4 \\
5 < x + 3 < 7 \\
|x + 3| < 7
\end{gather*}

\begin{align*}
|x^2 - 9| = |(x - 3)(x + 3)| < |x + 3||x - 3| < 7\delta \leqslant 7\cdot\frac{\epsilon}{7}= \epsilon,
\end{align*}
as required.
\end{itemize}

\item Using $\epsilon$-$\delta$ argument to prove
\begin{align*}
\lim_{x\to a}x^2 = a^2
\end{align*}
\item Alternative way to find $\delta$: Let $\epsilon > 0$ be given.
\begin{itemize}

-\epsilon  < x^2 - a^2 < \epsilon \\
a^2 - \epsilon < x^2 < a^2 + \epsilon \\

0 < x < \sqrt{a^2 + \epsilon} \\
- a < x - a < \sqrt{a^2 + \epsilon} - a \\
\end{gather*}

For any $x,y \in \mathbb{R}$

|x+y| \leqslant |x| + |y|
\end{align*}
\end{lemma}
\begin{proof}
\begin{itemize}
\item (First Proof) Note that $|x| = \max\{x, -x\}$ and $\pm x\leqslant |x|$. From
\begin{align*}
x + y \leqslant |x| + y \leqslant |x| + |y|, \\
-x - y \leqslant |x| - y \leqslant |x| + |y|,
\end{align*}
$|x + y|\leqslant |x| + |y|$.
\item (Second Proof) Note that $\forall x\in\mathbb{R}$, $x^2 = |x|^2$ and $x \leqslant |x|$. From
\begin{align*}
|x + y|^2 = (x + y)^2 = x^2 + 2 x y + y^2 \leqslant |x|^2 + 2 |x||y| + |y|^2 = (|x| + |y|)^2,
\end{align*}
$|x + y|\leqslant |x| + |y|$.
\end{itemize}
\end{proof}
From
\begin{align*}
|x| = |y + (x - y)| &\leqslant |y| + |x - y| \\
|y| = |x + (y - x)| &\leqslant |x| + |y - x|
\end{align*}
We have
\begin{align*}
||x| - |y|| \leqslant |x - y|
\end{align*}
\begin{lemma}\label{lem fx F1}
If $\ds \lim_{x \to a} f(x) = F$, $a$, $F\in\mathbb{R}$, then $\exists\,\delta >0$ such that when $0<|x-a|<\delta$ we have $|f(x)| < |F| + 1$.
\end{lemma}
\begin{proof}
Take $\epsilon = 1$, by definition of $\ds \lim_{x \to a} f(x) = F$,
\begin{align*}
\exists\,\delta > 0\,\,(0 < |x - a| < \delta\,\,\Longrightarrow\,\, |f(x) - F| < 1)
\end{align*}
From the formula $||x| - |y|| \leqslant |x - y|$,
\begin{align*}
||f(x)| - |F|| \leqslant |f(x) - F| < 1
\end{align*}
Hence
\begin{align*}
-1 < |f(x)| - |F| < 1
\end{align*}
So $|f(x)| < |F| + 1$.
\end{proof}
\begin{lemma}\label{lem fx F2}
If $\ds \lim_{x \to a} f(x) = F$, $a$, $F\in\mathbb{R}$ with $F\ne 0$, then $\exists\,\delta >0$ such that when $0 < |x-a| < \delta$ we have $|f(x)|> \frac{|F|}{2}$.
\end{lemma}
\begin{proof}
Take $\epsilon = \frac{|F|}{2}$, by definition of $\ds \lim_{x \to a} f(x) = F$,
\begin{align*}
\exists\,\delta > 0\,\,(0 < |x - a| < \delta\,\,\Longrightarrow\,\, |f(x) - F| < \frac{|F|}{2})
\end{align*}
\begin{itemize}
\item
From the formula $||x| - |y|| \leqslant |x - y|$,
\begin{align*}
||f(x)| - |F|| \leqslant |f(x) - F| < \frac{|F|}{2}
\end{align*}
Hence
\begin{align*}
-\frac{|F|}{2} < |f(x)| - |F| < \frac{|F|}{2}
\end{align*}
So $\frac{|F|}{2} < |f(x)|$.
\end{itemize}
\end{proof}
\begin{theorem}[Uniqueness of Limit]
Let $a, F_1, F_2 \in \mathbb{R}$ with
\begin{align*}
\lim_{x \to a} f(x) = F_1 \text{ and }\lim_{x \to a} f(x) = F_2
\end{align*}
then $F_1 = F_2$.
\end{theorem}
\begin{proof}
If $F_1\ne F_2$, set $\epsilon = \frac{|F_1 - F_2|}{3}$.
\begin{itemize}
\item By $\lim_{x\to a}f(x) = F_1$, $$\exists\,\delta_1 > 0\,\,(0 < |x - a| < \delta_1\,\Longrightarrow\,|f(x) - F_1| < \epsilon)$$
\item By $\lim_{x\to a}f(x) = F_2$, $$\exists\,\delta_2 > 0\,\,(0 < |x - a| < \delta_2\,\Longrightarrow\,|f(x) - F_2| < \epsilon)$$
\item Take $\delta = \min\{\delta_1, \delta_2\}$, then for $ 0 < |x - a| < \delta$,
\begin{align*}
3\epsilon = |F_1 - F_2| = |(f(x) - F_2) - (f(x) - F_1)| \leqslant |f(x) - F_2| + |f(x) - F_1| < \epsilon + \epsilon = 2\epsilon
\end{align*}
$3 < 2$, a contradiction. Hence $F_1 = F_2$.
\end{itemize}
\end{proof}
\begin{theorem}[Theorem~\ref{thm arith lim} --- limit of a sum.]
If $\lim_{x\to a}f(x) = F$, $\lim_{x\to a}g(x) = G$ then
\begin{align*}
\lim_{x\to a} f(x) + g(x) = F + G.
\end{align*}
\end{theorem}
\begin{proof}
Let $\epsilon > 0$ be given.
\begin{itemize}
\item By $\lim_{x\to a}f(x) = F$, $$\exists\,\delta_1 > 0\,\,\Big(0 < |x - a| < \delta_1\,\Longrightarrow\,|f(x) - F| < \frac{\epsilon}{2}\Big)$$
\item By $\lim_{x\to a}g(x) = G$, $$\exists\,\delta_2 > 0\,\,\Big(0 < |x - a| < \delta_2\,\Longrightarrow\,|g(x) - G| < \frac{\epsilon}{2}\Big)$$
\item Take $\delta = \min\{\delta_1, \delta_2\}$, then for $ 0 < |x - a| < \delta$,
\begin{align*}
|f(x) + g(x) - (F + G)| &= |(f(x) - F) + (g(x) - G)| \\
&\leqslant |f(x) - F| + |g(x) - G| \\
&< \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon.
\end{align*}
\end{itemize}
\end{proof}
\begin{theorem}[Theorem~\ref{thm arith lim} --- limit of a product.]
If $\lim_{x\to a}f(x) = F$, $\lim_{x\to a}g(x) = G$ then
\begin{align*}
\lim_{x\to a} f(x)\cdot g(x) = F\cdot G.
\end{align*}
\end{theorem}
\begin{proof}
Let $\epsilon > 0$ be given.
\begin{itemize}
\item By $\lim_{x\to a}f(x) = F$, $$\exists\,\delta_1 > 0\,\,\Big(0 < |x - a| < \delta_1\,\Longrightarrow\,|f(x) - F| < \frac{\epsilon}{2(|G| + 1)}\Big)$$
\item By $\lim_{x\to a}f(x) = F$, $$\exists\,\delta_2 > 0\,\,\Big(0 < |x - a| < \delta_2\,\Longrightarrow\,|f(x)| < |F| + 1\Big)$$
\item By $\lim_{x\to a}g(x) = G$, $$\exists\,\delta_3 > 0\,\,\Big(0 < |x - a| < \delta_3\,\Longrightarrow\,|g(x) - G| < \frac{\epsilon}{2(|F| + 1)}\Big)$$
\item Take $\delta = \min\{\delta_1, \delta_2, \delta_3\}$, then for $ 0 < |x - a| < \delta$,
\begin{align*}
|f(x)\cdot g(x) - F\cdot G| &= |f(x)(g(x) - G) + G(f(x) - F)| \\
&\leqslant |f(x)|\,|g(x) - G| + |G|\,|f(x) - F|\\
&< (|F| + 1)\cdot\frac{\epsilon}{2(|F| + 1)} + |G|\,\frac{\epsilon}{2(|G| + 1)}\\
&< \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon.
\end{align*}
\end{itemize}
\end{proof}
\begin{theorem}[Theorem~\ref{thm arith lim} --- limit of a reciprocal]
If $\lim_{x\to a}g(x) = G$, $G\ne 0$, then
\begin{align*}
\lim_{x\to a}\frac{1}{g(x)} = \frac{1}{G}.
\end{align*}
\end{theorem}
\begin{proof}
Let $\epsilon > 0$ be given.
\begin{itemize}
\item By $\lim_{x\to a}g(x) = G$, $$\exists\,\delta_1 > 0\,\,\Big(0 < |x - a| < \delta_1\,\Longrightarrow\,|g(x) - G| < \frac{|G|^2}{2}\epsilon\Big)$$
\item By $\lim_{x\to a}g(x) = G$, $$\exists\,\delta_2 > 0\,\,\Big(0 < |x - a| < \delta_2\,\Longrightarrow\,|g(x)| > \frac{|G|}{2}\Big)$$
\item Take $\delta = \min\{\delta_1, \delta_2\}$, then for $ 0 < |x - a| < \delta$,
\begin{align*}
\bigg|\frac{1}{g(x)} - \frac{1}{G}\bigg| &= \bigg|\frac{g(x) - G}{g(x)G}\bigg| \\
&\leqslant \frac{\frac{|G|^2}{2}\epsilon}{\frac{|G|}{2}|G|} = \epsilon.
\end{align*}
\end{itemize}
\end{proof}
\begin{theorem}[Theorem~\ref{thm squeeze} --- Squeeze theorem] %(or sandwich theorem or pinch theorem)]
Let $a \in \mathbb{R}$ and let $f, g, h$ be functions s.t.%three functions so that
\begin{align*}
f(x) \leqslant g(x) \leqslant h(x)
\end{align*}
for all $x$ in an interval around $a$, except possibly at $x=a$. If
\begin{align*}
\lim_{x \to a} f(x) &= \lim_{x \to a} h(x) = L
\end{align*}
then %it is also the case that
\begin{align*}
\lim_{x \to a} g(x) &= L
\end{align*}
\end{theorem}
\begin{proof}
Let $\epsilon > 0$ be given.
\begin{itemize}
\item By $\lim_{x\to a}f(x) = L$, $$\exists\,\delta_1 > 0\,\,(0 < |x - a| < \delta_1\,\Longrightarrow\,|f(x) - L| < \epsilon\,\Longrightarrow\, L - \epsilon < f(x) < L + \epsilon)$$
\item By $\lim_{x\to a}h(x) = L$, $$\exists\,\delta_2 > 0\,\,(0 < |x - a| < \delta_2\,\Longrightarrow\,|h(x) - L| < \epsilon\,\Longrightarrow\, L - \epsilon < h(x) < L + \epsilon)$$
\item Take $\delta = \min\{\delta_1, \delta_2\}$, then for $ 0 < |x - a| < \delta$,
\begin{align*}
L - \epsilon < f(x)\leqslant g(x)\leqslant h(x) < L + \epsilon
\end{align*}
Hence
\begin{align*}
L - \epsilon < g(x) < L + \epsilon\,\,\Longrightarrow\,\,|g(x) - L| < \epsilon
\end{align*}
\end{itemize}
\end{proof}
\begin{theorem}[Theorem~\ref{thm comp cont}]
If $f$ is continuous at $b$ and $\ds \lim_{x \to a} g(x) = b$ then $\ds \lim_{x\to a} f(g(x)) = f(b)$. I.e.
\begin{align*}
\lim_{x \to a} f\big(g(x)\big) &= f\big(\lim_{x \to a} g(x)\big)
\end{align*}
\end{theorem}
\begin{proof}
Let $\epsilon > 0$ be given. We want to show that
$$\exists\,\delta > 0\,\,(0 < |x - a| < \delta\,\Longrightarrow\,|f(g(x)) - f(b)| < \epsilon)$$
\begin{itemize}
\item By the continuity of $f$ at $b$, $\lim_{y\to b}f(y) = f(b)$. So
\begin{align}\label{eq:cont}
\exists\,\delta_1 > 0\,\,(0 < |y - b| < \delta_1\,\Longrightarrow\,|f(y) - f(b)| < \epsilon)
\end{align}
\item By $\lim_{x\to a}g(x) = b$, $$\exists\,\delta_2 > 0\,\,(0 < |x - a| < \delta_2\,\Longrightarrow\,|g(x) - b| < \delta_1)$$
\item Take $\delta = \delta_2$ and let $y = g(x)$, then for $ 0 < |x - a| < \delta$ $\Longrightarrow$ $|g(x) - b| < \delta_1$, by \eqref{eq:cont} $|f(g(x)) - f(b)| < \epsilon$, as required.
\end{itemize}
\end{proof}

\begin{defn}[Derivative at a point]\label{def:DIFFderiv}
Let $a\in\bbbr$ and let $f(x)$ be defined on an open interval
that contains $a$.
\begin{itemize}
\item The derivative of $f(x)$ at $x=a$ is denoted $f'(a)$ and is defined by
\begin{align*}
f'(a)=\lim_{h\rightarrow 0}\frac{f\big(a+h\big)-f(a)}{h}
\end{align*}
if the limit exists.
\item When the above limit exists, the function $f(x)$ is said to be differentiable
at $x=a$. When the limit does not exist, the function $f(x)$ is said to be not
differentiable at $x=a$.
\item We can equivalently define the derivative $f'(a)$ by the limit
\begin{align*}
f'(a)=\lim_{x\rightarrow a}\frac{f(x)-f(a)}{x-a}.
\end{align*}
\end{itemize}
\end{defn}
\begin{eg}[Derivative of $f(x)=c$]\label{eg:DIFFderivX0}
Let $a, c \in \bbbr$ be a constants. Compute the derivative of %the constant function
$f(x) = c$ at $x=a$.
\begin{align*}

&= \lim_{h \to 0} 0 &&\text{(simplified things)}\\
&= 0
\end{align*}

\begin{theorem}[Linearity of differentiation]\label{thm:DIFFlinearity}
Let $f(x),g(x)$ be differentiable, $\alpha, \beta \in \mathbb{R}$ be constants and define the  ``linear combination''
\begin{align*}
S(x) &= \alpha f(x) + \beta g(x).
\end{align*}
Then the derivative of $S(x)$ at $x=a$ exists and is
\begin{align*}
\diff{S}{x} = S'(x) &= \alpha f'(x) + \beta g'(x).
\end{align*}
\end{theorem}
\begin{theorem}[The product rule]\label{thm:DIFFprodRule}
Let $f(x),g(x)$ be differentiable, then %the derivative of the product $f(x)g(x)$ exists and is given by
\begin{align*}
\diff{}{x} \big\{ f(x) \, g(x) \big\} &= f'(x) \, g(x) + f(x) \, g'(x).
\end{align*}
\end{theorem}
\begin{cor}[Derivative of a square]\label{cor_2_4}

\begin{theorem}[The quotient rule]\label{thm:DIFFquotRule}

\diff{}{x} \left\{ \frac{f(x)}{g(x)} \right\} = \frac{f'(x) \, g(x) - f(x) \, g'(x)}{g(x)^2}.

\subsection*{Proof of the Linearity of Differentiation (Theorem \ref{thm:DIFFlinearity})}
\begin{align*}
\diff{}{x} \big\{\alpha\,f(x) + \beta\,g(x)\big\} &= \lim_{h\to 0}\frac{(\alpha\,f(x+h) + \beta\,g(x+h)) - (\alpha\,f(x) + \beta\,g(x))}{h} \\
&=\lim_{h\to 0}\frac{\alpha\,(f(x+h) - f(x)) + \beta\,(g(x+h)-g(x))}{h} \\
&=\lim_{h\to 0}\alpha\,\frac{f(x+h) - f(x)}{h} + \beta\,\frac{g(x+h)-g(x)}{h} \\
&= \alpha\,f'(x) + \beta\,g'(x)
\end{align*}
\subsection*{Proof of the Product Rule (Theorem \ref{thm:DIFFprodRule})}
\begin{align*}
\diff{}{x} \big\{f(x)g(x)\big\} &= \lim_{h\to 0}\frac{f(x+h)g(x+h) - f(x)g(x)}{h} \\
&=\lim_{h\to 0}\frac{f(x+h)g(x+h) - f(x)g(x+h) + f(x)g(x+h) - f(x)g(x)}{h} \\
&=\lim_{h\to 0}\frac{(f(x+h)- f(x))g(x+h) + f(x)(g(x+h) - g(x))}{h} \\
&=\lim_{h\to 0}\frac{f(x+h)- f(x)}{h}g(x+h) + f(x)\frac{g(x+h) - g(x)}{h} \\
&= f'(x)\, g(x) + f(x)\,g'(x)
\end{align*}
\subsection*{Proof of the Quotient Rule (Theorem \ref{thm:DIFFquotRule})}
\begin{align*}
\diff{}{x} \left\{ \frac{f(x)}{g(x)} \right\} &= \lim_{h\to 0}\frac{\frac{f(x+h)}{g(x+h)} - \frac{f(x)}{g(x)}}{h} =\lim_{h\to 0}\frac{\frac{f(x+h)g(x) - f(x)g(x+h)}{g(x+h)g(x)}}{h} \\
&=\lim_{h\to 0}\frac{\frac{f(x+h)g(x) - f(x)g(x) + f(x)g(x) - f(x)g(x+h)}{g(x+h)g(x)}}{h} \\
&=\lim_{h\to 0}\frac{\frac{(f(x+h) - f(x))\,g(x) - f(x)\,(g(x + h) - g(x))}{g(x+h)g(x)}}{h} \\
&=\lim_{h\to 0}\frac{\frac{f(x+h) - f(x)}{h}g(x) - f(x)\frac{g(x + h) - g(x)}{h}}{g(x+h)g(x)} \\
&= \frac{f'(x) \, g(x) - f(x) \, g'(x)}{g(x)^2}
\end{align*}

\begin{theorem}[The chain rule  --- version 1]\label{thm:DIFFchainRuleV1}
Let $a \in \mathbb{R}$, $g(x)$ be differentiable at $x=a$, $f(u)$ be differentiable at $u=g(a)$. Then $F(x) = f(g(x))$ is differentiable at $x=a$ and
\begin{align*}
F'(a) &=f'\big(g(a)\big)\,g'(a)
\end{align*}
\end{theorem}
\begin{theorem}[The chain rule --- version 2]\label{thm:DIFFchainRuleV2}
Let $f$ and $g$ be differentiable, then
\begin{align*}
\diff{}{x} f\big( g( x) \big) &= f'\big(g(x)\big) \cdot g'(x)
\end{align*}
\end{theorem}
\begin{theorem}[The chain rule --- version 3]\label{thm:DIFFchainRuleV3}
Let $y = f(u)$ and $u = g(x)$ be differentiable, then
\begin{align*}
\diff{y}{x} &= \diff{y}{u} \cdot \diff{u}{x}
\end{align*}
\end{theorem}

\subsection*{Derivation of the Chain Rule}
\begin{itemize}
\item Straightforward but flawed:
\begin{align*}
\big(f(g(x))\big)' &= \lim_{h\to 0}\frac{f(g(x + h)) - f(g(x))}{h} = \lim_{h\to 0}\frac{f(g(x+h)) - f(g(x))}{g(x + h) - g(x)}\cdot\frac{g(x + h) - g(x)}{h} \\
& ??\,=\lim_{h\to 0}\frac{f(g(x+h)) - f(g(x))}{g(x + h) - g(x)}\cdot\lim_{h\to 0}\frac{g(x + h) - g(x)}{h} = f'(g(x))\cdot g'(x)
\end{align*}
\item For $f$ is differentiable at $u = g(x)$ and $g$ is differentiable at $x$, set
\begin{align}\label{eq:E(k)}
E(k) = \frac{f(u + k) - f(u)}{k} - f'(u), \qquad E(0) = 0
\end{align}
Then $E(k)$ is continuous at $k = 0$ from
\begin{align*}
\lim_{k\to 0}E(k) = f'(u) - f'(u) = 0 = E(0)
\end{align*}
From \eqref{eq:E(k)}
\begin{align*}
f(u + k) - f(u) = k\big(E(k) + f'(u)\big)
\end{align*}
Set $u = g(x)$ and $k = g(x + h) - g(x)$, then $u + k = g(x + h)$; for $g$ is differentiable at $x$,
\begin{align*}
\lim_{h\to 0}k = \lim_{h\to 0} g(x + h) - g(x) = 0.
\end{align*}
So
\begin{align*}
\big(f(g(x))\big)' &= \lim_{h\to 0}\frac{f(g(x + h)) - f(g(x))}{h} = \lim_{h\to 0}\big(f'(g(x)) + E(k)\big)\cdot\frac{g(x + h) - g(x)}{h} \\
&\,=\lim_{h\to 0}\big(f'(g(x)) + E(k)\big)\cdot\lim_{h\to 0}\frac{g(x + h) - g(x)}{h} = f'(g(x))\cdot g'(x)
\end{align*}
\end{itemize}

\begin{theorem}[Rolle's theorem]\label{thm:DIFFrolle}
Let $a, b\in\mathbb{R}$ with $a<b$ and $f$ be a function so that
\begin{itemize}
\item $f(x)$ is continuous on $[a, b]$
\item $f(x)$ is differentiable on $(a, b)$
\item $f(a) = f(b)$
\end{itemize}
then $\exists c\in (a, b)$ with $f'(c)=0$.
\end{theorem}

\begin{cor}[Consequences of the mean value theorem]\label{cor:DIFFmvtcons}
Let $A,B\in\mathbb{R}$ with $A<B$; $f(x)$ be defined and continuous on $[A, B]$ and differentiable on $(A, B)$.
\begin{enumerate}[(a)]
\item If $f'(c)=0$ for $A<c<B$, then $f(b)=f(a)$ for all $A\leqslant a<b\leqslant B$.%\\
\item If $f'(c)\ge 0$ for $A<c<B$, then $f(b)\ge f(a)$ for $A\leqslant a\leqslant b\leqslant B$.%\\
\item If $f'(c)\leqslant 0$ for $A<c<B$, then $f(b)\leqslant f(a)$ for $A\leqslant a \leqslant b\leqslant B$.%\\
\end{enumerate}
\end{cor}
\begin{cor}\label{cor equal diff}
If $f'(x) = g'(x)$ on $(a, b)$, then $f-g$ is constant on $(a,b)$. %That is $f(x)=g(x)+c$, where $c$ is some constant.
\end{cor}

\subsection{First Approximation --- the Linear approximation}\label{ssec_first_approx}
\begin{impeqn}[Linear approximation]\label{eq:linApprox}
\begin{align*}
f(x) \approx f(a)+f'(a)(x-a)

Use the linear approximation to estimate $e^{0.1}$.
\begin{itemize}

\item To form the linear approximation, we need $f(a)=f(0)$ and $f'(a) = f'(0)$:
\begin{align*}
f(x) = e^x \;\Longrightarrow\; f(0) = 1\\
f'(x) = e^x \;\Longrightarrow\; f'(0) = 1
\end{align*}
\item The linear approximation of $f(0.1) = e^{0.1}$ is
\begin{align*}
\widetilde{f}(x) = f(0) + xf'(0) = 1 + x \;\Longrightarrow\; \widetilde{f}(0.1)  = 1 + 0.1 = 1.1
\end{align*}

\end{itemize}

f(x) = e^x \;\Longrightarrow\; f(0) = 1\\
f'(x) = e^x \;\Longrightarrow\; f'(0) = 1\\

\begin{align*}

\begin{defn}[Taylor polynomial]\label{def_3_4_2}
Let $a\in\mathbb{R}$ and $n\in\mathbb{Z}^+$. The $n^\mathrm{th}$ degree / order Taylor polynomial for $f(x)$ about $x=a$ is
\begin{align*}
T_n(x) = \sum_{k=0}^n \frac{f^{(k)}(a)}{k!}\cdot (x-a)^k.
\end{align*}
The special case $a=0$ is called a Maclaurin polynomial.%\footnote{The polynomials are named after
\end{defn}
\begin{gather*}
\text{To make the first $n$ derivatives of $T_n(x)$ at $x = a$ match those of $f(x)$} \\
T_n(a) = f(a),\quad T_n'(a) = f'(a),\quad T_n''(a) = f''(a),\quad\ldots,\quad T_n^{(n)}(a) = f^{(n)}(a)\\
T_n(x) = c_0 + c_1 (x - a) + c_2 (x - a)^2 + \cdots + c_n (x - a)^n \\
c_0 = f(a),\quad c_1 = f'(a),\quad c_2 = \frac{f''(a)}{2!},\quad\ldots,\quad c_n = \frac{f^{(n)}(a)}{n!}
\end{gather*}

\subsection{The Error in the Taylor Polynomial Approximations}\label{ssec taylor error}
\begin{theorem}[Taylor Theorem]\label{eq:taylorErrorN}
Let $f$ be defined on an open interval $I$, $a, b\in I$, and $f^{(n+1)}$ exists on $I$, $n\in\mathbb{Z}^+$. Then $\exists\;c$ lies between $a$ and $b$ s.t.
\begin{align*}
f(b) = f(a) + f'(a) (b - a) + \frac{f''(a)}{2!}\cdot(b-a)^2 + \cdots + \frac{f^{(n)}(a)}{n!}\cdot(b-a)^n + \frac{f^{(n+1)}(c)}{(n+1)!}\cdot(b-a)^{n+1}
\end{align*}
\end{theorem}
\begin{itemize}
\item Since differentiability implies continuity, $f$, $f'$, $f''$, $\ldots$, $f^{(n)}$ are all continuous on $I$.
\item WLOG, let $a < b$.
\item Let $M$, $g(t)$ be defined s.t.
\begin{align*}
f(b) &= f(a) + f'(a)(b - a) + \frac{f''(a)}{2!}(b - a)^2 +\cdots+\frac{f^{(n)}(a)}{n!}(b-a)^n + \frac{M}{(n + 1)!}(b-a)^{n+1} \\
g(t) &= -f(b) + f(t) + f'(t)(b - t) + \frac{f''(t)}{2!}(b - t)^2 +\cdots+\frac{f^{(n)}(t)}{n!}(b-t)^n + \frac{M}{(n + 1)!}(b-t)^{n+1}
\end{align*}
\item Note that $g(a) = 0$, $g(b) = 0$, then by Rolle's Theorem, $\exists\,c\in(a, b)$ s.t. $g'(c) = 0$.
\begin{align*}
g'(t) &= f'(t) - f'(t) + f''(t)(b-t) - f''(t)(b-t) + \frac{f'''(t)}{2!}(b - t)^2 - \frac{f'''(t)}{2!}(b - t)^2 + \cdots \\ &= \frac{f^{(n+1)}(t) - M}{n!}(b-t)^n
\end{align*}
\item So $g'(c) = 0$ implies $f^{(n+1)}(c) - M = 0\,\,\Longrightarrow\,\, M = f^{(n+1)}(c)$.
\end{itemize}
\begin{eg}[Showing $e<3$]\label{eg_3_4_5}
Using \eqref{eq:taylorErrorN}, show that $e < 3$.
\begin{align*}
e^1 = 1 + 1 + \frac{1}{2} + \frac{e^c}{6}, \quad c\in(0, 1)
\end{align*}
\end{eg}
\begin{itemize}
\item Prove that $e$ is irrational.
\item Take $a=0$, $b=1$ and $f(x) = e^x$ in Taylor's Theorem, $\exists\;c\in(0, 1)$ s.t.
\begin{align*}
&f(1) = f(0) + f'(0) + \frac{f''(0)}{2!} + \cdots + \frac{f^{(n)}(0)}{n!} + \frac{f^{(n+1)}(c)}{(n+1)!} \\
&\Longrightarrow\; e - \sum_{k = 0}^n \frac{1}{k!} = \frac{e^c}{(n+1)!}
\end{align*}
from above
\begin{align*}
\frac{1}{(n+1)!} < e - \sum_{k = 0}^n \frac{1}{k!} < \frac{3}{(n+1)!}
\end{align*}
\item If $e$ is rational, then $\exists\;n\geqslant 3$ s.t. $e\cdot n!\in\mathbb{N}$, so
\begin{align*}
\frac{n!}{(n+1)!} < e\cdot n! - \sum_{k = 0}^n \frac{n!}{k!} < \frac{3\cdot n!}{(n+1)!} \leqslant\frac{3}{4}
\end{align*}
But both $e\cdot n!$ and $\sum_{k = 0}^n \frac{n!}{k!}$ are integers, the difference of these two integer is a positive number not exceeding $\frac{3}{4}$, a contradiction !
\end{itemize}

\begin{defn}\label{def:APPlocalMaxMin} %%%%%%%%%%%%%%%%%%%%%
Let $f(x)$ be defined on an interval $I$ and $c\in I$. Then
\begin{itemize}
\item $f(x)$ has a \emph{global (absolute) minimum on $I$} at $x=c$ if $f(x)\geqslant f(c)\;\forall\,x\in I$.
\item $f(x)$ has a \emph{global (absolute) maximum on $I$} at $x=c$ if $f(x)\leqslant f(c)\;\forall\,x\in I$.
\item $f(x)$ has a \emph{local minimum on $I$} at $x=c$ if $\exists\,h_0>0$, $f(x)\geqslant f(c)\;\forall\,x\in I\land|x - c|<h_0$.
\item $f(x)$ has a \emph{local maximum on $I$} at $x=c$ if $\exists\,h_1>0$, $f(x)\leqslant f(c)\;\forall\,x\in I\land|x - c|<h_1$.
\end{itemize}
\begin{align*}
\text{global maxima and minima / global extrema} \\
\text{local maxima and minima / local extrema}
\end{align*}
\end{defn}

\end{center}

\begin{theorem}\label{thm:APPlocalMaxMin}
Let $f(x)$ be defined on an interval $I$ and $a,b,c\in I$ with $a<c<b$. If $f(x)$ has a local minimum or local maximum at $x=c$ and $f'(c)$ exists, then $f'(c)=0$.
\end{theorem}
\begin{itemize}
\item If $f(x)$ has a local minimum at $x=c$, then $\exists\,h_0>0$, $f(x) - f(c)\geqslant 0$ $\;\forall\,x\in I\land|x - c|<h_0$.
\begin{itemize}
\item Take such $x$ with $x < c$, then
\begin{align*}
\frac{f(x) - f(c)}{x - c} \leqslant 0 \;\Longrightarrow\; f'(c) = \lim_{x\to c-}\frac{f(x) - f(c)}{x - c}\leqslant 0
\end{align*}
\item Take such $x$ with $x > c$, then
\begin{align*}
\frac{f(x) - f(c)}{x - c} \geqslant 0 \;\Longrightarrow\; f'(c) = \lim_{x\to c+}\frac{f(x) - f(c)}{x - c}\geqslant 0
\end{align*}
\end{itemize}
So $f'(c) = 0$.
\item If $f(x)$ has a local maximum at $x=c$, then $\exists\,h_1>0$, $f(x) - f(c)\leqslant 0$ $\;\forall\,x\in I\land|x - c|<h_1$.
\begin{itemize}
\item Take such $x$ with $x < c$, then
\begin{align*}
\frac{f(x) - f(c)}{x - c} \geqslant 0 \;\Longrightarrow\; f'(c) = \lim_{x\to c-}\frac{f(x) - f(c)}{x - c}\geqslant 0
\end{align*}
\item Take such $x$ with $x > c$, then
\begin{align*}
\frac{f(x) - f(c)}{x - c} \leqslant 0 \;\Longrightarrow\; f'(c) = \lim_{x\to c+}\frac{f(x) - f(c)}{x - c}\leqslant 0
\end{align*}
\end{itemize}
So $f'(c) = 0$.
\end{itemize}
\begin{theorem}\label{thm:APPsecondDerivTest}
Let $f(x)$ be defined on an interval $I$ and $a,b,c\in I$ with $a<c<b$. If $f'(c) = 0$ and
\begin{itemize}
\item $f''(c) < 0$, then $f(x)$ has a local maximum at $x = c$.
\item $f''(c) > 0$, then $f(x)$ has a local minimum at $x = c$.
\end{itemize}
\end{theorem}
\begin{itemize}
\item If $f''(c) < 0$,
\begin{align*}
f''(c) = \lim_{h\to 0}\frac{f'(c + h) - f'(c)}{h} = \lim_{h\to 0}\frac{f'(c + h)}{h} < 0
\end{align*}
Then
\begin{itemize}
\item $f'(c + h) < 0$ for sufficiently small positive $h$: $f$ is decreasing at the right of $x = c$
\item $f'(c + h) > 0$ for sufficiently small negative $h$: $f$ is increasing at the left of $x = c$
\end{itemize}
So $f(x)$ has a local maximum at $x = c$.
\item If $f''(c) > 0$,
\begin{align*}
f''(c) = \lim_{h\to 0}\frac{f'(c + h) - f'(c)}{h} = \lim_{h\to 0}\frac{f'(c + h)}{h} > 0
\end{align*}
Then
\begin{itemize}
\item $f'(c + h) > 0$ for sufficiently small positive $h$: $f$ is increasing at the right of $x = c$
\item $f'(c + h) < 0$ for sufficiently small negative $h$: $f$ is decreasing at the left of $x = c$
\end{itemize}
So $f(x)$ has a local minimum at $x = c$.
\end{itemize}
\begin{defn}\label{def:APPcriticalPoint} %%%%%%%%%%%%%%%%%%%%%
Let $f(x)$ be defined on $a<x<b$ and $a<c<b$. Then
\begin{itemize}
\item if $f'(c)$ exists and is zero, then $x=c$ is a \emph{critical} point
\item if $f'(c)$ does not exist, then $x=c$ is a \emph{singular} point.
\end{itemize}
\end{defn}

\subsection{Finding Global Maxima and Minima}\label{ssec find maxmin}
Let $f(x)$ be continuous on $[a, b]$. Two most important theorems:
\begin{align*}
&\text{1.(IVT) If $Y$ is any number between $f(a)$ and $f(b)$ then $\exists\,c\in[a,b]$ s.t. $f(c) = Y$.}\\
&\text{2.(Max-Min) $f$ attains the global extremum on $[a, b]$.}
\end{align*}
\begin{theorem}\label{thm:APPglobalMaxMinExist}
Let $f(x)$ be continuous on the closed, finite interval $-\infty<a\leqslant x\leqslant b<\infty$.
Then $f(x)$ attains a maximum and a minimum at least once. That is, there exist numbers $a\leqslant x_m, x_M\leqslant b$ such that
\begin{align*}
f(x_m)\leqslant f(x)\leqslant f(x_M)\qquad\text{for all }a\leqslant x\leqslant b
\end{align*}
\end{theorem}
\subsection*{Existence of Extremum on Open Intervals}
\begin{itemize}
\item Let $f(x)$ is continuous on $(a, b)$ and
\begin{align*}
\lim_{x\to a+}f(x) = L,\quad\lim_{x\to b-}f(x) = M.
\end{align*}
\begin{enumerate}
\item If $\exists\,u\in(a, b)$ s.t. $f(u) > L$ and $f(u) > M$, then $f$ has a global maximum value on $(a, b)$.
\item If $\exists\,u\in(a, b)$ s.t. $f(u) < L$ and $f(u) < M$, then $f$ has a global minimum value on $(a, b)$.
\end{enumerate}
Here $a$ may be $-\infty$, $b$ may be $\infty$, $L$, $M$ can be $-\infty$, $\infty$.
\item Case 1.
\begin{itemize}
\item By $\displaystyle \lim_{x\to a+}f(x) = L$, given $\varepsilon > 0$, $\exists\,\delta_1 > 0$ s.t. for $a < x < a + \delta_1$, $f(x) < L + \varepsilon$.
\item Take $\varepsilon = f(u) - L$, then
\begin{align}\label{joi}
\exists\,x_1\in (a, u) \text{ s.t. for } a < x < x_1, \,f(x) < L + (f(u) - L) = f(u).
\end{align}
\item By $\displaystyle \lim_{x\to b-}f(x) = M$, given $\varepsilon > 0$, $\exists\,\delta_2 > 0$ s.t. for $b - \delta_2 < x < b$, $f(x) < M + \varepsilon$.
\item Take $\varepsilon = f(u) - M$, then
\begin{align}\label{jio}
\exists\,x_2\in (u, b)\text{ s.t. for } x_2 < x < b, \,f(x) < M + (f(u) - M) = f(u).
\end{align}
\item Apriori $f$ is continuous on $[x_1, x_2]$, so $f$ has global maxinum value on $[x_1, x_2]$, say at $w\in[x_1, x_2]$. But $u\in[x_1, x_2]\;\Longrightarrow\;f(w)\geqslant f(u)$; together with \eqref{joi}, \eqref{jio}, $f$ has a global maximum value on $w\in(a, b)$.
\end{itemize}
\item Case 2.
\begin{itemize}
\item By $\displaystyle \lim_{x\to a+}f(x) = L$, given $\varepsilon > 0$, $\exists\,\delta_1 > 0$ s.t. for $a < x < a + \delta_1$, $f(x) > L - \varepsilon$.
\item Take $\varepsilon = L - f(u)$, then
\begin{align}\label{joiw}
\exists\,x_1\in (a, u) \text{ s.t. for } a < x < x_1, \,f(x) > L - (L - f(u)) = f(u).
\end{align}
\item By $\displaystyle \lim_{x\to b-}f(x) = M$, given $\varepsilon > 0$, $\exists\,\delta_2 > 0$ s.t. for $b - \delta_2 < x < b$, $f(x) > M - \varepsilon$.
\item Take $\varepsilon = M - f(u)$, then
\begin{align}\label{jiow}
\exists\,x_2\in (u, b)\text{ s.t. for } x_2 < x < b, \,f(x) > M - (M - f(u)) = f(u).
\end{align}
\item Apriori $f$ is continuous on $[x_1, x_2]$, so $f$ has global mininum value on $[x_1, x_2]$, say at $w\in[x_1, x_2]$. But $u\in[x_1, x_2]\;\Longrightarrow\;f(w)\leqslant f(u)$; together with \eqref{joiw}, \eqref{jiow}, $f$ has a global minimum value on $w\in(a, b)$.
\end{itemize}
\end{itemize}
\begin{theorem}\label{thm:APPglobalMaxMin}
If $f(x)$ has a global maximum or global minimum for $a\leqslant x\leqslant b$ at $x=c$, then
\begin{itemize}
\item $f'(c)=0$, or
\item $f'(c)$ does not exist, or
\item $c=a$ or $c=b$.
\end{itemize}
\end{theorem}
\begin{cor}\label{cor find maxmin}
Let $f(x)$ be defined on $a \leqslant x \leqslant b$. To find the global extremum of $f$ is to find the extremal values of $f$ at all the critical, singular, and end points.
\end{cor}

\subsection*{Concavity and Inflections, Sketching the Graph of a Function}
\begin{itemize}
\item Definition of Concavity: Let $f:I\mapsto\mathbb{R}$ be differentiable.
\begin{itemize}
\item $f$ is concave up on $I$ if $f'$ is increasing on $I$.
\item $f$ is concave down on $I$ if $f'$ is decreasing on $I$.
\end{itemize}
\item Definition of inflection points: $x_0$ is an inflection point of $y = f(x)$ if
\begin{itemize}
\item $y = f(x)$ has a tangent line at $x = x_0$
\item the concavity of $f$ is opposite on opposite sides of $x_0$
\end{itemize}
\item Theorem: Let $f:I\mapsto\mathbb{R}$ be differentiable.
\begin{enumerate}
\item If $f'' > 0$ on $I$, then $f$ is concave up on $I$.
\item If $f'' < 0$ on $I$, then $f$ is concave down on $I$.
\item If $f$ has an inflection on $x_0\in I$, then $f''(x_0) = 0$.
\end{enumerate}
\end{itemize}

\section{L'H\^opital's Rule and Indeterminate Forms}\label{sec_3_7}
\begin{align*}
\lim_{x\to a}f(x) = F, \quad\lim_{x\to a}g(x) = G\ne 0 \quad\Longrightarrow\quad \lim_{x\to a}\frac{f(x)}{g(x)} = \frac{F}{G}
\end{align*}
\begin{itemize}
\item Examples of
\begin{align*}
\lim_{x \to 0} f(x) = 0, \quad \lim_{x \to 0} g(x) = 0\quad\Longrightarrow\quad\lim_{x \to 0} \frac{f(x)}{g(x)} = \text{undetermined}
\end{align*}
\begin{align*}
f(x)&=x & g(x)&=x^2 & \lim_{x\to0} \frac{x}{x^2}
&= \lim_{x\to0} \frac{1}{x} = \text{DNE} \\
f(x)&=x^2 & g(x)&=x & \lim_{x\to0} \frac{x^2}{x}
&= \lim_{x\to0} x  = 0 \\
f(x)&=x & g(x)&=x & \lim_{x\to0} \frac{x}{x}
&= \lim_{x\to0} 1 = 1  \\
f(x)&=7x^2 & g(x)&=3x^2 & \lim_{x\to0} \frac{7x^2}{3x^2}
&= \lim_{x\to0} \frac{7}{3} = \frac{7}{3} \\
\end{align*}
\item Examples of
\begin{align*}
\lim_{x \to 0} f(x) = +\infty, \quad \lim_{x \to 0} g(x) = +\infty\quad\Longrightarrow\quad\lim_{x \to 0} \frac{f(x)}{g(x)} = \text{undetermined}
\end{align*}
\begin{align*}
f(x)&=\frac{1}{x^4} & g(x)&=\frac{1}{x^6} & \lim_{x\to0} \frac{\frac{1}{x^4}}{\frac{1}{x^6}}
&= \lim_{x\to0}x^2 = 0 \\
f(x)&=\frac{1}{x^4} & g(x)&=\frac{1}{x^4} & \lim_{x\to0} \frac{\frac{1}{x^4}}{\frac{1}{x^4}}
&= \lim_{x\to0} 1  = 1 \\
f(x)&=\frac{1}{x^4} & g(x)&=\frac{1}{x^2}& \lim_{x\to0} \frac{\frac{1}{x^4}}{\frac{1}{x^2}}
&= \lim_{x\to0}\frac{1}{x^2} = +\infty
\end{align*}
\end{itemize}
\stepcounter{theorem}
\begin{theorem}[L'H\^opital's Rule]\label{thm:APPlhopital}
\begin{itemize}
\item $\big(\frac{0}{0}\text{ type}\big)$ If
\begin{itemize}
\item $\exists\,\delta > 0$ s.t. $\forall\,x\in(a, a + \delta)$, $f(x)$, $g(x)$ are differentiable and $g(x)\ne 0$.
\item $\ds\lim_{x\to a+}f(x) = \lim_{x\to a+}g(x) = 0$.
\item $\ds\lim_{x\to a+}\frac{f'(x)}{g'(x)} = l\in\overline{\mathbb{R}}$.
\end{itemize}
Then $\ds\lim_{x\to a+}\frac{f(x)}{g(x)} = l$.
\item $\big(\frac{\infty}{\infty}\text{ type}\big)$ If
\begin{itemize}
\item $\exists\,\delta > 0$ s.t. $\forall\,x\in(a, a + \delta)$, $f(x)$, $g(x)$ are differentiable.
\item $\ds\lim_{x\to a+}|f(x)| = \lim_{x\to a+}|g(x)| = \infty$.
\item $\ds\lim_{x\to a+}\frac{f'(x)}{g'(x)} = l\in\overline{\mathbb{R}}$.
\end{itemize}
Then $\ds\lim_{x\to a+}\frac{f(x)}{g(x)} = l$.
\end{itemize}
Here $a+$ could be $a-$, $a$; $a$ could be $\pm\infty$ (with some minor modification).
\end{theorem}

\begin{align*}
\lim\limits_{x\to a}\frac{f(x)}{g(x)} & \text{ exists.}

\begin{align*}
a=0,\quad f(x)=x^2\sin\frac{1}{x}, \quad g(x)= x.
\end{align*}
then
\begin{gather*}
\lim_{x\to 0} f(x) = 0\quad\text{and}\quad\lim_{x\to 0} g(x) = 0. \\

\end{gather*}

\lim_{x\to 0}\frac{f(x)}{g(x)} &= \lim_{x\to 0}\frac{x^2\sin\frac{1}{x}}{x} =\lim_{x\to 0} x\sin\frac{1}{x}= 0\qquad\text{By squeeze theorem}
\end{align*}
\end{warning}

- \underbrace{\frac{1}{\log(1+x)}}_{\rightarrow\pm\infty}\Big)
=\lim_{x\rightarrow 0}\underbrace{\frac{\log(1+x)-x}{x\log(1+x)}}_{\atp
{\mathrm{num}\rightarrow 0}
{\mathrm{den}\rightarrow 0}}
\tag{E1}

\lim_{x\rightarrow 0}\underbrace{\frac{\log(1+x)-x}{x\log(1+x)}}_{\atp
{\mathrm{num}\rightarrow 0}
{\mathrm{den}\rightarrow 0}}
&=\lim_{x\rightarrow 0}\frac{\frac{1}{1+x}-1} {\log(1+x)+\frac{x}{1+x}}
=\lim_{x\rightarrow 0}\frac{1-(1+x)} {(1+x)\log(1+x)+x}\notag\\
&=-\lim_{x\rightarrow 0}\underbrace{\frac{x} {(1+x)\log(1+x)+x}}_{\atp
{\mathrm{num}\rightarrow 0}
{\mathrm{den}\rightarrow 1\times 0+0=0}} \tag{E2}

{\mathrm{num}\rightarrow 0}

=-\lim_{x\rightarrow 0}\underbrace{\frac{1} {\log(1+x)+\frac{1+x}{1+x}+1}}_{\atp

=-\frac{1}{2}
\tag{E3}
\end{equation}
