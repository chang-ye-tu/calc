    binomial & $\ds X\sim {\sf bin}(n, p)$ & $\ds {\sf p}(x) = \binom{n}{x}\,p^x(1 - p)^{n - x},\; x = 0,\,\ldots,\,n$ & $\ds np$ & $\ds np(1 - p)$ \\ [30pt]
    Poisson & $\ds X\sim {\sf poi}(\lambda)$ & $\ds {\sf p}(x) = \frac{e^{-\lambda}\lambda^x}{x!},\; x\geqslant 0$ & $\ds\lambda$ & $\ds\lambda$ \\ [30pt]
    geometric & $\ds X\sim {\sf geo}(p)$ & $\ds {\sf p}(x) = p(1 - p)^{x - 1},\; x = 1,\,2,\,\ldots$ & $\ds\frac{1}{p}$ & $\ds\frac{1 - p}{p^2}$ \\ [30pt]
    negative binomial & $\ds X\sim {\sf nbin}(r, p)$ & $\ds {\sf p}(x) = \binom{x - 1}{r - 1}\,p^r(1 - p)^{x - r},\; x\geqslant r$ & $\ds\frac{r}{p}$ & $\ds\frac{r(1 - p)}{p^2}$ \\ [30pt]
    hypergeometric & \makecell{$\ds X\sim {\sf hgeo}(n, N, m)$ \\ $\ds p = \frac{m}{N}$} & $\ds {\sf p}(x) = \frac{\binom{m}{x}\binom{N - m}{n - x}}{\binom{N}{n}},\; x = 0,\,\ldots,\,m$ & $\ds np$ & $\ds\frac{N - n}{N - 1}\cdot np(1 - p)$ \\ %[30pt]
    \bottomrule
  \end{tabular}
  }

    \bottomrule
  \end{tabular}
