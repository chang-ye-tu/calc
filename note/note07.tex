\subsection*{Тдѓт┐хУѕЄтЪ║ТюгТђДУ│ф}

С║їжЄЇуЕЇтѕє $\approx$  (тИХугдУЎЪ) жФћуЕЇ: $z$ У╗ИСИіТќ╣уѓ║ТГБ, СИІТќ╣уѓ║У▓а. 

\begin{dfn}
  ухдт«џ $\Omega = [a,\,b]\times[c,\,d]$, $b\geqslant a$, $d\geqslant c$, $f:\Omega\to\mathbb{R}$. 
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item $\Omega$ тѕєтЅ▓ $\ds\mathbb{P}: a = x_0 < x_1 < x_2 < \cdots < x_n = b,\;c = y_0 < y_1 < y_2 < \cdots < y_m = d$
    \item $\ds\Delta x_k = x_k - x_{k-1}$,$\;$ $\ds\Delta y_l = y_l - y_{l - 1}$,$\;$ $k = 1,\,2,\,\ldots,\,n$;$\;$ $l = 1,\,2,\,\ldots,\,m$
    \item $\ds\|\mathbb{P}\| = \max\,\{\Delta x_k\Delta y_l\;|\;1\leqslant k\leqslant n,\;1\leqslant l\leqslant m\}$
    \item ТеБТюгж╗ъ $\ds(\xi_k,\zeta_l)$: $\ds x_{k - 1} \leqslant \xi_k \leqslant x_k$,$\;$ $\ds y_{l - 1} \leqslant \zeta_l \leqslant y_l$,$\;$ $k = 1,\,2,\,\ldots,\,n$;$\;$ $l=1,\,2,\,\ldots,\,m$.
    \item $\ds u_{k,l} = \sup\left\{f(x,y)\;|\;x_{k-1}\leqslant x\leqslant x_k,\;y_{l-1}\leqslant y\leqslant y_l\right\}$,$\;$ $\ell_{k, l} = \inf\left\{f(x,y)\;|\;x_{k-1}\leqslant x\leqslant x_k,\;y_{l-1}\leqslant y\leqslant y_l\right\}$,$\;$ $k=1,\,2,\,\ldots,\,n$;$\;$ $l=1,\,2,\,\ldots,\,m$.
    \item $\ds R(f,\mathbb{P}) = \sum_{l=1}^m\sum_{k=1}^n f(\xi_k,\zeta_l)\,\Delta x_k\Delta y_l$,$\;$ $\ds U(f,\mathbb{P}) = \sum_{l=1}^m\sum_{k=1}^n u_{k,l}\,\Delta x_k\Delta y_l$,$\;$ $\ds L(f,\mathbb{P}) = \sum_{l=1}^m\sum_{k=1}^n \ell_{k, l}\,\Delta x_k\Delta y_l$; \\жА»уёХ $\ds L(f,\mathbb{P})\leqslant R(f,\mathbb{P})\leqslant U(f, \mathbb{P})$. 
    \item Т▒ѓ $\ds\lim_{\|\mathbb{P}\|\to 0} R(f,\mathbb{P})$. УІЦт░ЇСИЇтљїтѕєтЅ▓УѕЄТеБТюгж╗ъжЂИтЈќТГцТЦхжЎљтЮЄтГўтюеСИћуЏИуГЅ, уе▒ $f$ тюе $\Omega$ тЈ»уЕЇ (тѕє); $f(x, y)$ тюе $\Omega$ уџёт«џуЕЇтѕєуѓ║ $\ds\int_\Omega f(x, y)\,\mathrm{d}A \equiv \lim_{\|\mathbb{P}\|\to 0} R(f,\mathbb{P})$.
  \end{itemize}
\end{dfn}

\begin{prp}
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item[]
    \item УІЦ $f$ тюе $\Omega$ ТюЅуЋї, СИћжЎцС║є $\Omega$ СИГТюЅжЎљтђІт╣│Т╗ЉТЏ▓уиџтцќ $f$ тюе $\Omega$ жђБу║ї, тЅЄ $f$ тюе $\Omega$ тЈ»уЕЇтѕє. 
    \item УІЦ $\Omega = \Omega_1\cap\Omega_2$ СИћ $\Omega_1,\;\Omega_2$ тЮЄуѓ║уЪЕтйб, $f$ тюе $\Omega_1$, $\Omega_2$ тЮЄтЈ»уЕЇ, тЅЄ $\ds\int_\Omega f\,\text{d}A = \int_{\Omega_1}f\,\text{d}A + \int_{\Omega_2}f\,\text{d}A$.
    \item УІЦ $f_1,\;f_2$ тЮЄтюе $\Omega$ тЈ»уЕЇСИћ $f_1(x, y)\leqslant f_2(x, y)\;\forall\,(x,y)\in\Omega$, тЅЄ $\ds\int_\Omega f_1\,\text{d}A \leqslant\int_\Omega f_2\,\text{d}A$.
    \item УІЦ $\alpha,\,\beta\in\mathbb{R}$, $f_1,\;f_2$ тЮЄтюе $\Omega$ тЈ»уЕЇ, тЅЄ $\ds\int_\Omega(\alpha f_1 + \beta f_2)\,\text{d}A = \alpha\!\int_\Omega f_1\,\text{d}A + \beta\!\int_\Omega f_2\,\text{d}A$.
  \end{itemize}
\end{prp}

\subsection*{жђљТгАуЕЇтѕє}

\subsubsection*{уЪЕтйбуЕЇтѕєтЇђтЪЪ}

\begin{thm}[Fubini]
  УІЦ $\Omega = [a,\,b]\times[c,\,d]$, $f:\Omega\to\mathbb{R}$ уѓ║тЈ»уЕЇ, тЅЄ $$\int_\Omega f(x, y)\,\text{d}A = \int_c^d\bigg\{\int_a^b f(x, y)\,\text{d}x\bigg\}\,\text{d}y = \int_a^b\bigg\{\int_c^d f(x, y)\,\text{d}y\bigg\}\,\text{d}x$$
\end{thm}

\begin{ex}
  С╗ц $\Omega = \{(x,\,y)\,|\,-1\leqslant x\leqslant 1,\;-2\leqslant y\leqslant 2\}$, тЅЄ $\ds\int_\Omega\sqrt{1 - x^2}\,\text{d}A = \frac{\pi}{2}\times 4 = 2\pi$.
\end{ex}

\begin{ex}
  Т▒ѓ $\ds\int_0^2\!\int_1^3 x^2y\,\text{d}y\,\text{d}x$ УѕЄ $\ds\int_1^3\!\int_0^2 x^2y\,\text{d}x\,\text{d}y$.
\end{ex}

\begin{sol}
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item[]
    \item $\ds\int_0^2\!\int_1^3 x^2y\,\text{d}y\,\text{d}x = \int_0^2 x^2\Big(\frac{y^2}{2}\,\Big|_1^3\Big)\,\text{d}x = \int_0^2 x^2\Big(\frac{9}{2} - \frac{1}{2}\Big)\,\text{d}x = \frac{4}{3}x^3\,\Big|_0^2 = \frac{32}{3}$.
    \item $\ds\int_1^3\!\int_0^2 x^2y\,\text{d}x\,\text{d}y = \int_1^3 y\Big(\frac{x^3}{3}\,\Big|_0^2\Big)\,\text{d}y = \int_1^3 y\,\frac{8}{3}\,\text{d}x = \frac{4}{3}y^2\,\Big|_1^3 = \frac{36}{3} - \frac{4}{3} = \frac{32}{3}$.
  \end{itemize}
\end{sol}

\begin{ex}
  УІЦ $\Omega = [0,\,1]\times[0,\,3]$, Т▒ѓ $\ds\int_\Omega e^{2x + y}\,\text{d}A$.
\end{ex}

\begin{sol}
  $\ds\int_\Omega e^{2x + y}\,\text{d}A = \int_0^3\!\int_0^1 e^{2x + y}\,\text{d}x\,\text{d}y = \int_0^3 e^y\bigg(\int_0^1 e^{2x}\,\text{d}x\bigg)\,\text{d}y = \int_0^3 e^y\Big(\frac{e^{2x}}{2}\,\Big|_0^1\Big)\,\text{d}y = \frac{(e^2 - 1)(e^3 - 1)}{2}$.
\end{sol}

\begin{ex}
  УІЦ $\Omega = [0,\,\pi]\times[0,\,2\pi]$, Т▒ѓ $\ds\int_\Omega\!\sin(x + y)\,\text{d}A$.
\end{ex}

\begin{sol}
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item[]
    \item $\ds\int_\Omega\!\sin(x + y)\,\text{d}A = \int_0^\pi\!\int_0^{2\pi}\!\sin(x + y)\,\text{d}y\,\text{d}x = \int_0^\pi\!\Big(-\cos(x + y)\,\Big|_0^{2\pi}\Big)\,\text{d}x = -\int_0^\pi\!(\cos(x + 2\pi) - \cos x)\,\text{d}x = 0$.
    \item $\ds\int_\Omega\!\sin(x + y)\,\text{d}A = \int_0^{2\pi}\!\int_0^{\pi}\!\sin(x + y)\,\text{d}x\,\text{d}y = \int_0^{2\pi}\!\Big(-\cos(x + y)\,\Big|_0^{\pi}\Big)\,\text{d}y = -\int_0^{2\pi}\!(\cos(\pi + y) - \cos y)\,\text{d}y = -\int_0^{2\pi}\!(\cos\pi\cos y - \sin\pi\sin y - \cos y)\,\text{d}y = 2\int_0^{2\pi}\!\cos y\,\text{d}y = 0$.
  \end{itemize}
\end{sol}

\begin{ex}
  УІЦ $\Omega = [1,\,2]\times[0,\,1]$, Т▒ѓ $\ds\int_\Omega ye^{xy}\,\text{d}A$.
\end{ex}

\begin{sol}
  \begin{itemize}\setlength{\itemsep}{0pt}
    \item[]
    \item $\ds\int_\Omega ye^{xy}\,\text{d}A = \int_0^1\!\int_1^2 ye^{xy}\,\text{d}x\,\text{d}y = \int_0^1 y\,\Big(\frac{e^{xy}}{y}\,\Big|_1^2\Big)\,\text{d}y = \int_0^1(e^{2y} - e^y)\,\text{d}y = \frac{e^2}{2} - e + \frac{1}{2}$. 
    \item $\ds\int_\Omega ye^{xy}\,\text{d}A = \int_1^2\!\int_0^1 ye^{xy}\,\text{d}y\,\text{d}x = \int_1^2\Big(y\,\frac{e^{xy}}{x} - \frac{e^{xy}}{x^2}\,\Big|_0^1\Big)\,\text{d}x = \int_1^2\Big(\frac{e^{x}}{x} - \frac{e^x}{x^2} + \frac{1}{x^2}\Big)\,\text{d}x = \Big(\frac{e^x}{x}-\frac{1}{x}\Big)\,\Big|_1^2 = \frac{e^2}{2} - e + \frac{1}{2}$.
