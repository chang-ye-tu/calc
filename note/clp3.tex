\section{Limits}\label{sec limits}
\begin{defn}[Limit]\label{def limit}
Let
\begin{itemize}\itemsep1pt \parskip0pt \parsep0pt
\item[$\circ$]
$m$, $n\in\mathbb{N}$ %be natural numbers\footnote{In this text, we will interested in $m,n\in\big\{1,2,3\big\}$, but the definition works for all natural numbers $m,n$.}
\item[$\circ$]
$\va\in\bbbr^m$
\item[$\circ$]
$f(\vx)$ be defined for all $\vx$ near %\footnote{To be precise, there is a number $r>0$ such that $f(\vx)$ is defined for all $\vx$ obeying $|\vx-\va|<r$.}
$\va$ and take values in $\bbbr^n$
\item[$\circ$]
$\vL\in\bbbr^n$
\end{itemize}
We write
\begin{align*}
\lim_{\vx\to \va} f(\vx) = \vL
\end{align*}
if %\footnote{There is a precise, formal version of this definition that looks just like Definition 1.7.1 of the CLP-1 text.}
$f(\vx)$ is arbitrarily close to $\vL$ whenever $\vx$ is close enough to $\va$,
without %\footnote{You may find the condition ``without  being exactly $\va$''
being exactly $\va$.
\end{defn}
Example: Given $\vx = (x_1, x_2)$, $\va = (1, 1)$, $\vL = (2,\,1,\,\sin 2)$, and $f:\mathbb{R}^2\mapsto\mathbb{R}^3$: $f(\vx)\equiv f(x_1, x_2) = (x_1 + x_2,\,{x_1}^2{x_2}^2,\,\sin(x_1 + {x_2}^2))$. Then $\ds\lim_{\vx\to \va} f(\vx) = \vL$.
Focus on $n = 1$.
\begin{theorem}[Properties of Limits]\label{thm LIMpptiesB}
Let
\begin{itemize}%\itemsep1pt \parskip0pt \parsep0pt
\item[$\circ$]
$m$, $n\in\mathbb{N}$
\item[$\circ$]
$\va\in\bbbr^m$ and $\vb\in\bbbr^n$
\item[$\circ$]
$D\subseteq\bbbr^m$ that contains all $\vx\in\bbbr^m$ that are near $\va$
\item[$\circ$]
$c,F,G\in\bbbr$
\end{itemize}
and
\begin{equation*}
f,g:D\setminus\{\va\}\rightarrow\bbbr\qquad
\vX:\bbbr^m\setminus\{\vb\}\rightarrow D\setminus\{\va\}\qquad
\ga:\bbbr\rightarrow\bbbr
\end{equation*}
Assume that
\begin{equation*}
\lim_{\vx\to \va} f(\vx) = F\qquad
\lim_{\vx\to \va} g(\vx) = G\qquad
\lim_{\vy\to \vb} \vX(\vy) = \va\qquad
\lim_{t\to F} \ga(t) = \ga(F)
\end{equation*}
Then
\begin{enumerate}[(a)]
\item\qquad
$\lim\limits_{\vx\rightarrow \va}\big[f(\vx)+g(\vx)\big]
=F+G$\\
\null\qquad$\lim\limits_{\vx\rightarrow \va}\big[f(\vx)-g(\vx)\big]
=F-G$
\item\qquad
$\lim\limits_{\vx\rightarrow \va} f(\vx)\,g(\vx)
=FG$\\
\null\qquad$\lim\limits_{\vx\rightarrow \va} cf(\vx)
=cF$
\item\qquad
$\lim\limits_{\vx\rightarrow \va} \frac{f(\vx)}{g(\vx)}
=\frac{F}{G}$ if $G\ne 0$
\item\qquad
$\lim\limits_{\vy\rightarrow \vb} f\big(\vX(\vy)\big)
=F$
\item\qquad
$\lim\limits_{\vx\rightarrow \va} \ga\big(f(\vx)\big)
=\ga(F)$
\end{enumerate}
\end{theorem}
\begin{eg}\label{eg LIMtwodA}
Evaluate
\begin{equation*}
\lim_{(x,y)\rightarrow (2,3)} \frac{x+\sin y}{x^2y^2+1}
\end{equation*}
\soln
The following ``$\eqover{a}$'' means that part (a) of Theorem \ref{thm LIMpptiesB} justifies that equality.
\begin{align*}
\lim_{(x,y)\rightarrow (2,3)} \big(x+\sin y\big)\
&\eqover{a} \lim_{(x,y)\rightarrow (2,3)} x
+\lim_{(x,y)\rightarrow (2,3)}\sin y \\
&\eqover{e} \lim_{(x,y)\rightarrow (2,3)} x
+\sin\Big(\lim_{(x,y)\rightarrow (2,3)} y\Big) \\
&=\  2+\sin 3\\[0.05in]
\lim_{(x,y)\rightarrow (2,3)} \big(x^2y^2+1\big)\
&\eqover{a}\  \lim_{(x,y)\rightarrow (2,3)} x^2y^2
+\lim_{(x,y)\rightarrow (2,3)}1 \\
&\eqover{b}\  \Big(\lim_{(x,y)\rightarrow (2,3)} x\Big)
\Big(\lim_{(x,y)\rightarrow (2,3)} x\Big)
\Big(\lim_{(x,y)\rightarrow (2,3)} y\Big)
\Big(\lim_{(x,y)\rightarrow (2,3)} y\Big)+1 \\
&=\  2^23^2+1\\
\lim_{(x,y)\rightarrow (2,3)} \frac{x+\sin y}{x^2y^2+1}\
&\eqover{c}\ \frac
{\lim\limits_{(x,y)\rightarrow (2,3)}(x+\sin y)}
{\lim\limits_{(x,y)\rightarrow (2,3)}(x^2y^2+1)}\\[0.05in]
&=\ \frac{2+\sin 3}{37}
\end{align*}

\begin{eg}\label{ex LIMtwodB}
Evaluate $\ds\lim\limits_{(x,y)\rightarrow (0,0)}\frac{x^2y}{x^2+y^2}$.
\soln
\begin{align*}
\frac{x^2y}{x^2+y^2} =\frac{(r\cos\theta)^2(r\sin\theta)}{r^2} = r\cos^2\theta\sin\theta
\end{align*}
As $\big|r\cos^2\theta\sin\theta\big|\leqslant r\to 0$ as $r\to 0$,
$$
\lim\limits_{(x,y)\rightarrow (0,0)}\frac{x^2y}{x^2+y^2}=0
$$
\end{eg}
\begin{eg}\label{ex LIMtwodC}
Evaluate $\ds\lim\limits_{(x,y)\rightarrow (0,0)}\frac{x^2-y^2}{x^2+y^2}$.
\soln
$$f(x,y)=\frac{x^2-y^2}{x^2+y^2}=\frac{(r\cos\theta)^2-(r\sin\theta)^2}{r^2}
=\cos^2\theta-\sin^2\theta
=\cos(2\theta)$$
$$\lim\limits_{(x,y)\rightarrow (0,0)}\frac{x^2-y^2}{x^2+y^2} = \text{DNE}$$
\end{eg}
\begin{eg}\label{eg nasty limit}
Study the behaviour of the function
$$f(x,y) =
\begin{cases}
\frac{(2x-y)^2}{x-y} & \text{if $x\ne y$}\\[0.05in]
0 & \text{if $x=y$}
\end{cases}$$
as $(x,y)\rightarrow (0,0)$.
\soln
\begin{itemize}
\item Along $y = x - x^3$
\begin{align*}
f(x,x-x^3)&=\frac{{(2x-x+x^3)}^2}{x-x+x^3}
=\frac{{(x+x^3)}^2}{x^3} \\
&=\frac{{(1+x^2)}^2}{x}\longrightarrow
\begin{cases}+\infty & \text{as $x\rightarrow 0$ with $x>0$}\\
\noalign{\vskip0.1in}
-\infty & \text{as $x\rightarrow 0$ with $x<0$}
\end{cases}
\end{align*}
\item Along $y = x - ax^2$, $a\not=0$
$$
\lim_{x\rightarrow 0}f(x,x-ax^2)
=\lim_{x\rightarrow 0}\frac{{(2x-x+ax^2)}^2}{x-x+ax^2}
=\lim_{x\rightarrow 0}\frac{{(x+ax^2)}^2}{ax^2}
=\lim_{x\rightarrow 0}\frac{{(1+ax)}^2}{a}
=\frac{1}{a}
$$
\item Conclusion:
$$\lim\limits_{(x,y)\rightarrow (0,0)}f(x, y) = \text{DNE}$$
\end{itemize}
\end{eg}
\section{Partial Derivatives}\label{sec partials}
\begin{equation*}
\diff{f}{x}(a) =\lim_{h\rightarrow 0}\frac{f(a+h) - f(a)}{h}
=\lim_{x\rightarrow a}\frac{f(x) - f(a)}{x-a}
\end{equation*}
\begin{equation*}
\diff{f}{x}(x) =\lim_{h\rightarrow 0}\frac{f(x+h) - f(x)}{h}
\end{equation*}
\begin{defn}[Partial Derivatives]\label{def partials}
The $x$- and $y$-partial derivatives of $f(x,y)$ are
\begin{align*}
\pdiff{f}{x}(x,y)
&= \lim_{h\rightarrow 0}\frac{f(x+h,y) - f(x,y)}{h} \\[0.05in]
\pdiff{f}{y}(x,y)
&= \lim_{h\rightarrow 0}\frac{f(x,y+h) - f(x,y)}{h}
\end{align*}
\end{defn}
\begin{notn}\label{notn partial}
The partial derivative $\ds\pdiff{f}{x}(x,y)$ of $f(x,y)$ is also denoted
\begin{equation*}
\pdiff{f}{x}\qquad
f_x(x,y)\qquad
f_x\qquad
D_xf(x,y)\qquad
D_xf\qquad
D_1 f(x,y)\qquad
D_1 f
\end{equation*}
$\ds\pdiff{f}{x}(a,b)$ is also denoted
$\ds\pdiff{f}{x}\bigg|_{(a,b)}:\;$ evaluated at $(x,y)=(a,b)$.
$\ds{\left(\pdiff{f}{x}\right)}_{\!y}$: $y$ is being held.%is used to make explicit that the variable $y$ is being held.
\end{notn}
\subsubsection{Evaluation of Partial Derivatives}\label{sec eval partials}
\begin{itemize}
\item
to evaluate $\pdiff{f}{x}(x,y)$, treat the $y$ in $f(x,y)$ as a constant
and differentiate the resulting function of $x$ with respect to $x$.
\item
To evaluate $\pdiff{f}{x}(a,b)$, treat the $y$ in $f(x,y)$ as a constant
and differentiate the resulting function of $x$ with respect to $x$.
Then evaluate the result at $x=a$, $y=b$.
\end{itemize}
\stepcounter{theorem}
\begin{eg}\label{eg partials A}
\begin{equation*}
f(x,y) = x^3+y^2+ 4xy^2
\end{equation*}
\begin{align*}
\pdiff{f}{x}%={\left(\pdiff{f}{x}\right)}_{\!y}
&= \pdiff{}{x}(x^3) + \pdiff{}{x}(y^2) +\pdiff{}{x}(4xy^2) \\
&= 3x^2+0 + 4y^2\pdiff{}{x}(x) \\
&= 3x^2 +4y^2
\end{align*}
\begin{align*}
\pdiff{f}{y}%={\left(\pdiff{f}{y}\right)}_{\!x}
&= \pdiff{}{y}(x^3) + \pdiff{}{y}(y^2) +\pdiff{}{y}(4xy^2) \\
&= 0 + 2y + 4x\pdiff{}{y}(y^2) \\
&= 2y+8xy
\end{align*}
\begin{alignat*}{3}
\pdiff{f}{x}(1,0) &= 3(1)^2 +4(0)^2&=3 \\[0.05in]
\pdiff{f}{y}(1,0) &= 2(0) +8(1)(0)\ &=0
\end{alignat*}
\end{eg}%\goodbreak
\begin{eg}\label{eg partials B}
\begin{equation*}
f(x,y) = y\cos x + xe^{xy}
\end{equation*}
$\pdiff{}{x} e^{yx}=y e^{yx}$ and
\begin{align*}
\pdiff{f}{x}(x,y) &= y\pdiff{}{x}(\cos x) + e^{xy}\pdiff{}{x}(x)
+x\pdiff{}{x}\big(e^{xy}\big)
\qquad\text{(by the product rule)} \\
&= -y\sin x + e^{xy} +xye^{xy} \\[0.05in]
\pdiff{f}{y}(x,y) &= \cos x\pdiff{}{y}(y) + x\pdiff{}{y}\big(e^{xy}\big) \\
&= \cos x + x^2e^{xy}
\end{align*}
\end{eg}
\begin{eg}\label{eg partials C}
\begin{equation*}
f(x,y,z,t) = x\sin(y+2z) +t^2e^{3y}\ln z
\end{equation*}
\begin{align*}
\pdiff{f}{x}(x,y,z,t) &= \sin(y+2z)  \\[0.05in]
\pdiff{f}{y}(x,y,z,t) &= x\cos(y+2z) +3t^2e^{3y}\ln z \\[0.05in]
\pdiff{f}{z}(x,y,z,t) &= 2x\cos(y+2z) +t^2e^{3y}/z \\[0.05in]
\pdiff{f}{t}(x,y,z,t) &= 2te^{3y}\ln z
\end{align*}
\end{eg}
\begin{eg}\label{eg partials D}
Set
\begin{equation*}
f(x,y)=\begin{cases}
\frac{\cos x-\cos y}{x-y}&\text{if $x\ne y$}\\
0&\text{if $x=y$}
\end{cases}
\end{equation*}
\begin{itemize}
\item For $x\ne y$, $\ds f_x = \frac{-\sin x(x - y) - (\cos x - \cos y)}{(x - y)^2}$. Cannot get $f_x(0, 0)$ by this.
\item Compute $f_x(0, 0)$ by definition:
\begin{align*}
f_x(0,0) &= \lim_{h\rightarrow 0}\frac{f(h,0)-f(0,0)}{h} \\
&= \lim_{h\rightarrow 0}\frac{\frac{\cos h-1}{h}-0}{h}
&\qquad\text{(Recall that $h\ne 0$ in the limit.)} \\
&= \lim_{h\rightarrow 0}\frac{\cos h-1}{h^2} \\
&= \lim_{h\rightarrow 0}\frac{-\sin h}{2h}
&\qquad\text{(By l'H\^opital's rule.)} \\
&= \lim_{h\rightarrow 0}\frac{-\cos h}{2}
&\qquad\text{(By l'H\^opital again.)} \\
&=-\frac{1}{2}
\end{align*}
\item
\begin{align*}
\lim_{(x, y)\to (0, 0)} \frac{\cos x - \cos y}{x - y} &= \lim_{(x, y)\to (0, 0)}\frac{-2\sin\frac{x+y}{2}\sin\frac{x-y}{2}}{x - y} \\
&= -\lim_{(x, y)\to (0, 0)}\sin\frac{x+y}{2}\,\lim_{(x, y)\to (0, 0)}\frac{\sin\frac{x-y}{2}}{\frac{x - y}{2}} = 0
\end{align*}
So $f(x,y)$ is continuous at $(0,0)$.
\item
$f(x, y)$ is not continuous at $(a, a)$ for $a\ne 0$: by above $\ds\lim_{(x, y)\to(a, a)}f(x, y) = \sin a$ but by definition $f(a, a) = 0$.
\item
$f_x(0, 0) = f_y(0, 0) = -\frac{1}{2}$.
\end{itemize}
\end{eg}

\section{Higher Order Derivatives}\label{sec higher order}
\begin{alignat*}{5}
\pdiff{}{x}\left(\pdiff{f}{x}\right)(x,y)
&=\frac{\partial^2 f}{\partial x^2}(x,y) &= f_{xx}(x,y) \\
\pdiff{}{y}\left(\pdiff{f}{x}\right)(x,y)
&=\frac{\partial^2\ f}{\partial y\partial x}(x,y) &= f_{xy}(x,y) \\
\pdiff{}{x}\left(\pdiff{f}{y}\right)(x,y)
&=\frac{\partial^2\ f}{\partial x\partial y}(x,y) &= f_{yx}(x,y) \\
\pdiff{}{y}\left(\pdiff{f}{y}\right)(x,y)
&=\frac{\partial^2 f}{\partial y^2}(x,y) &= f_{yy}(x,y)
\end{alignat*}
\begin{eg}\label{eg higher order A}
Let $f(x,y) = e^{my}\cos(nx)$. Then
\begin{align*}
f_x &=  -n e^{my}\sin(nx) &
f_y &=   m e^{my}\cos(nx) \\
f_{xx}  &=   -n^2 e^{my}\cos(nx) &
f_{yx} &=   -m n e^{my}\sin(nx) \\
f_{xy}  &=   -m n e^{my}\sin(nx) &
f_{yy} &=   m^2 e^{my}\cos(nx)
\end{align*}
\end{eg}
\begin{eg}\label{eg higher order B}
Let $f(x,y) = e^{\al x+\be y}$. Then
\begin{align*}
f_x &=  \al e^{\al x+\be y} &
f_y &=  \be e^{\al x+\be y} \\
f_{xx}  &= \al^2  e^{\al x+\be y} &
f_{yx} &=   \be \al e^{\al x+\be y} \\
f_{xy}  &=  \al \be  e^{\al x+\be y} &
f_{yy} &=  \be^2 e^{\al x+\be y}
\end{align*}
For any integers $m,n\geqslant 0$,
\begin{equation*}
\frac{\partial^{m+n} f}{\partial x^m\, \partial y^n}
= \al^m\be^n e^{\al x+\be y}
\end{equation*}
\end{eg}
\begin{eg}\label{eg higher order C}
If $f(x_1,x_2,x_3,x_4) = x_1^4\, x_2^3\, x_3^2\, x_4$, then
\begin{align*}
\frac{\partial^4\  f}
{\partial x_1\, \partial x_2\,\partial x_3\,\partial x_4}
&=  \frac{\partial^3 \ }
{\partial x_1\, \partial x_2\,\partial x_3}
\left( x_1^4\, x_2^3\, x_3^2\right)  \\
&=  \frac{\partial^2 \ }
{\partial x_1\, \partial x_2}
\left( 2\ x_1^4\, x_2^3\, x_3\right)  \\
&=  \pdiff{}{x_1}
\left( 6\ x_1^4\, x_2^2\, x_3\right)  \\
&=   24\ x_1^3\, x_2^2\, x_3
\end{align*}
and
\begin{align*}
\frac{\partial^4\ f}
{\partial x_4\, \partial x_3\,\partial x_2\,\partial x_1}
&=  \frac{\partial^3 \ }
{\partial x_4\, \partial x_3\,\partial x_2}
\left( 4 x_1^3\, x_2^3\, x_3^2\,x_4\right)  \\
&=  \frac{\partial^2 \ }
{\partial x_4\, \partial x_3}
\left( 12\ x_1^3\, x_2^2\, x_3^2\,x_4\right)  \\
&=  \pdiff{}{x_4}
\left( 24\ x_1^3\, x_2^2\, x_3\,x_4\right)  \\
&=   24\ x_1^3\, x_2^2\, x_3
\end{align*}
\end{eg}
\begin{theorem}[Clairaut-Young-Schwarz: The Equality of Mixed Partials]
If %the partial derivatives
$\frac{\partial^2\hfil f\hfil\,}{\partial x\partial y}$ and
$\frac{\partial^2\hfil f\hfil\,}{\partial y\partial x}$ exist
and are continuous at $(x_0,y_0)$, then
$$
\frac{\partial^2\hfil f\hfil\,}{\partial x\partial y}(x_0,y_0)
=\frac{\partial^2\hfil f\hfil\,}{\partial y\partial x}(x_0,y_0)
$$
\end{theorem}
\section{The Chain Rule}\label{sec chain rule}
1-d chain rule:
\begin{equation*}
\diff{}{t}f\big(x(t)\big)
= \diff{f}{x}\big(x(t)\big)\ \diff{x}{t}(t)
\end{equation*}
\begin{theorem}[The Chain Rule]\label{thm:chainRule}
For $F(s,t)=f\big(x(s,t)\,,\,y(s,t)\big)$
\begin{align*}
\pdiff{F}{s}(s,t)
&=  \pdiff{f}{x}\big(x(s,t)\,,\,y(s,t)\big)\, \pdiff{x}{s}(s,t)
+\pdiff{f}{y}\big(x(s,t)\,,\,y(s,t)\big)\, \pdiff{y}{s}(s,t) \\
\pdiff{F}{t}(s,t)
&=  \pdiff{f}{x}\big(x(s,t)\,,\,y(s,t)\big)\, \pdiff{x}{t}(s,t)
+\pdiff{f}{y}\big(x(s,t)\,,\,y(s,t)\big)\, \pdiff{y}{t}(s,t)
\end{align*}
\end{theorem}%\noindent

\stepcounter{theorem}
\stepcounter{theorem}
\begin{eg}[$\pdiff{}{s}f\big(x(s,t),y(s,t)\big)$]\label{eg:chainRuleAA}
\begin{equation*}
f(x,y)=e^{xy}\qquad x(s,t)=s \qquad y(s,t)=\cos t
\end{equation*}
Define $F(s,t)=f\big(x(s,t)\,,\,y(s,t)\big)$; compute $\pdiff{F}{s}(s,t)$.
\soln
\begin{equation*}
\pdiff{F}{s}(s,t)
=  \pdiff{f}{x}\big(x(s,t)\,,\,y(s,t)\big)\, \pdiff{x}{s}(s,t)
+\pdiff{f}{y}\big(x(s,t)\,,\,y(s,t)\big)\, \pdiff{y}{s}(s,t)
\end{equation*}
\begin{align*}
f(x,y)&=e^{xy}  \\
\pdiff{f}{x}(x,y)&= ye^{xy}  &
\pdiff{f}{x}\big(x(s,t),y(s,t)\big)&= y(s,t) e^{x(s,t)\,y(s,t)}
= \cos t\ e^{s\cos t} \\
\pdiff{f}{y}(x,y)&= xe^{xy}  &
\pdiff{f}{y}\big(x(s,t),y(s,t)\big)&= x(s,t) e^{x(s,t)\,y(s,t)}
= s\ e^{s\cos t} \\
\pdiff{x}{s}&=1 &
\pdiff{y}{s}&=0
\end{align*}
$$
\pdiff{F}{s}(s,t)=\overbrace{\left\{\cos t\ e^{s\cos t}\right\}}^{\pdiff{f}{x}}
\overbrace{(1) }^{\pdiff{x}{s}}
+\overbrace{\left\{s\ e^{s\cos t}\right\}}^{\pdiff{f}{y}}
\overbrace{(0) }^{\pdiff{y}{s}}
= \cos t\ e^{s\cos t}
$$
\end{eg}
\begin{eg}[$\diff{}{t}f\big(x(t),y(t)\big)$]\label{eg:chainRuleA}
Find $\diff{}{t}f\big(x(t),y(t)\big)$ for
\begin{equation*}
f(x,y)=x^2-y^2\qquad x(t)=\cos t \qquad y(t)=\sin t
\end{equation*}
\soln
Define $F(t)=f\big(x(t),y(t)\big)$, then
$$
\diff{F}{t}(t)=
\pdiff{f}{x}\big(x(t),y(t)\big) \diff{x}{t}(t)
+\pdiff{f}{y}\big(x(t),y(t)\big) \diff{y}{t}(t)
$$
\begin{align*}
f(x,y)&=x^2-y^2  \\
\pdiff{f}{x}(x,y)&= 2x  &
\pdiff{f}{x}\big(x(t),y(t)\big)&= \phantom{-}2x(t) =\phantom{-}2\cos t  \\
\pdiff{f}{y}(x,y)&= -2y  &
\pdiff{f}{y}\big(x(t),y(t)\big)&= -2y(t)  =-2\sin t \\
\diff{x}{t}&=-\sin t &
\diff{y}{t}&=\cos t
\end{align*}
$$
\diff{F}{t}(t)=(2\cos t)(-\sin t) +(-2\sin t)(\cos t) = -4\sin t\cos t
$$
Of course, we can compute $F(t)$ explicitly:
$$
F(t)=f\big(x(t),y(t)\big)=x(t)^2-y(t)^2=\cos^2t-\sin^2t
$$
$$
F'(t)=2(\cos t)(-\sin t)-2(\sin t)(\cos t) = -4 \sin t\cos t
$$
\end{eg}
\begin{eg}[$\pdiff{}{t}f(x+ct)$]\label{eg:chainRuleB}
Define
$u(x,t)=x+ct$ and $w(x,t)=f(x+ct)=f\big(u(x,t)\big)$. Then
$$
\pdiff{}{t}f(x+ct)
=\pdiff{w}{t}(x,t)
=\diff{f}{u}\big(u(x,t)\big)  \pdiff{u}{t}(x,t)

\begin{theorem}[Tangent Plane and Normal Line]\label{thm tan plane G}
Let $(x_0,y_0,z_0)$ be on $G(x,y,z)=0$. Assume that the gradient
\begin{equation*}
\vnabla G(x_0,y_0,z_0)
= \llt \pdiff{G}{x}\big( x_0\,,\,y_0\,,\,z_0\big)\,,\,
\pdiff{G}{y}\big( x_0\,,\,y_0\,,\,z_0\big)\,,\,
\pdiff{G}{z}\big( x_0\,,\,y_0\,,\,z_0\big)\rgt
\end{equation*}
of $G$ at $(x_0,y_0,z_0)$ is nonzero.
\begin{enumerate}[(a)]
\item
$\vnabla G(x_0,y_0,z_0)$ is normal to $G(x,y,z)=0$ at $(x_0,y_0,z_0)$.
\item
The equation of the tangent plane to $G(x,y,z)=0$ at $(x_0,y_0,z_0)$ is
\begin{equation*}
\vnabla G(x_0,y_0,z_0)\cdot\llt x-x_0\,,\,y-y_0\,,\,z-z_0\rgt =0
\end{equation*}
\item
The equation of the normal line to $G(x,y,z)=0$ at $(x_0,y_0,z_0)$ is
\begin{equation*}
\llt x,y,z \rgt =  \llt x_0,y_0,z_0 \rgt
+t\,\vnabla G(x_0,y_0,z_0)
\end{equation*}
\end{enumerate}
\end{theorem}
Then normal vector $\vn$ at a point $\vr(t_0) = (x(t_0), y(t_0), z(t_0))$ on the surface $G(x, y, z) = 0$: for all curves on $G$ that pass through $\vr(t_0)$, $\vn\cdot\vr'(t_0) = 0$.
$\ds\frac{\text{d}}{\text{d}t}G(x(t), y(t), z(t)) = 0\,\Longrightarrow\,\frac{\partial G}{\partial x}\,\frac{\text{d}x}{\text{d}t} + \frac{\partial G}{\partial y}\,\frac{\text{d}y}{\text{d}t} + \frac{\partial G}{\partial z}\,\frac{\text{d}z}{\text{d}t} = 0\,\Longrightarrow\,\nabla G(\vr(t_0))\cdot\vr'(t_0) = 0$
\stepcounter{theorem}

\section{Maximum and Minimum Values}\label{sec max}
\begin{defn}\label{def local max min}
Let $f(x,y)$ be defined in $R\subseteq\bbbr^2$ and $(a,b)\in R$.
\begin{itemize}
\item
$(a,b)$ is a \emph{local maximum} of $f(x,y)$ if $f(x,y)\leqslant f(a,b)$ for all $(x,y)$ close to $(a,b)$.
More precisely, $(a,b)$ is a local maximum of $f(x,y)$ if there is an $r>0$ such that $f(x,y)\leqslant f(a,b)$ for
all points $(x,y)$ within a distance $r$ of $(a,b)$.
\item
$(a,b)$ is a \emph{local minimum} of $f(x,y)$ if $f(x,y)\geqslant f(a,b)$ for all $(x,y)$ close to $(a,b)$.
\item
Local maximum and minimum values are also called extremal values.
\item
$(a,b)$ is an \emph{absolute maximum} or \emph{global maximum} of $f(x,y)$ if $f(x,y)\leqslant f(a,b)$ $\forall\,(x,y)\in R$.
\item
$(a,b)$ is an \emph{absolute minimum} or \emph{global minimum} of $f(x,y)$ if $f(x,y)\geqslant f(a,b)$ $\forall\,(x,y)\in R$.
\end{itemize}
\end{defn}
\subsection*{Local Maxima and Minima}
\subsubsection*{1D}
If the largest value of $f(x)$ is $f(a)$ and $f$ is differentiable at $a$, then $f'(a)=0$.
\begin{nfig}
\begin{center}
\includegraphics{localMaxA.pdf}
\end{center}
\end{nfig}
Suppose that the largest value of $f(x)$ is $f(a)$. Then $\forall\,h>0$,
\begin{equation*}
f(a+h)\leqslant f(a)
\implies f(a+h)-f(a)\leqslant 0
\implies \frac{f(a+h)-f(a)}{h}\leqslant 0\quad\text{if $h>0$}
\end{equation*}
Taking the limit $h\rightarrow 0$ tells us that $f'(a)\leqslant 0$.
Similarly
\begin{equation*}
f(a+h)\leqslant f(a)
\implies f(a+h)-f(a)\leqslant 0
\implies \frac{f(a+h)-f(a)}{h}\geqslant 0\quad\text{if $h<0$}
\end{equation*}
Taking the limit $h\rightarrow 0$, $f'(a)\geqslant 0$. So we have both $f'(a)\geqslant 0$ and $f'(a)\leqslant 0$ which forces $f'(a)=0$.
\subsubsection*{2D}
Suppose that $f(x,y)$ is defined $\forall\,(x,y)\in R\subseteq\bbbr^2$ and $(a,b)$ is not on the boundary of $R$. Let $f$ has a local maximum at $(a,b)$.
\begin{efig}
\begin{center}
\includegraphics{max.pdf}
\end{center}
\end{efig}
Then $f(x,y)$ must decrease in value as $(x,y)$ moves away from $(a,b)$ in \emph{any} direction.
No matter which direction $\vd$ we choose, the directional derivative of $f$ at $(a,b)$ in direction $\vd$ must be zero or smaller.
\begin{align*}
D_\vd f(a,b) = \vnabla f(a,b)\cdot\frac{\vd}{|\vd|}\leqslant 0
\end{align*}
And the directional derivative of $f$ at $(a,b)$ in the direction $-\vd$ also must be zero or negative:
\begin{align*}
D_{-\vd} f(a,b) = \vnabla f(a,b)\cdot\frac{-\vd}{|\vd|}\leqslant 0
\quad\text{which implies that}\quad \vnabla f(a,b)\cdot\frac{\vd}{|\vd|}\geqslant 0
\end{align*}
\begin{theorem}\label{thm stationary point}
Let $f(x,y)$ be defined $\forall\,(x,y)\in R\subseteq\bbbr^2$. Assume that
\begin{itemize}\itemsep1pt \parskip0pt \parsep0pt
\item[$\circ$]
$(a,b)\in R$ that is not on the boundary of $R$ and
\item[$\circ$]
$(a,b)$ is a local maximum or local minimum of $f$ and that
\item[$\circ$]
the partial derivatives of $f$ exist at $(a,b)$.
\end{itemize}
Then
\begin{equation*}
\vnabla f(a,b) = \vZero.
\end{equation*}
\end{theorem}
\begin{defn}\label{def critical point}
Let $f(x,y)$ be a function and let $(a,b)$ be a point in its domain. Then
\begin{itemize}
\item if $\vnabla f(a,b)$ exists and is zero we call $(a,b)$
a \emph{critical} point (or a \emph{stationary} point) of the function, and
\item if $\vnabla f(a,b)$ does not exist then we call $(a,b)$
a \emph{singular} point of the function.
\end{itemize}
\end{defn}

\subsection*{The Second Derivative Test}
\begin{defn}\label{def saddle point}
The critical point $(a,b)$ is called a saddle point for $f(x,y)$ if $\forall\,r>0$,
\begin{itemize}
\item
$\exists\,(x,y)$, within a distance $r$ of $(a,b)$, s.t. $f(x,y)>f(a,b)$
\item
$\exists\,(x,y)$, within a distance $r$ of $(a,b)$, s.t. $f(x,y)<f(a,b)$.
\end{itemize}
\end{defn}
\begin{theorem}[Second Derivative Test]\label{thm second deriv test}
Let $r>0$ and assume that all second order derivatives of the function $f(x,y)$ are continuous at all points $(x,y)$ that are within a distance $r$ of $(a,b)$. Assume that $f_x(a,b)=f_y(a,b)=0$.
Define
\begin{equation*}
D(x,y) = f_{xx}(x,y)\,f_{yy}(x,y) - f_{xy}(x,y)^2
\end{equation*}
Then
\begin{itemize}
\item
if $D(a,b)>0$ and $f_{xx}(a,b)>0$, then $f(x,y)$ has a local minimum at $(a,b)$,
\item
if $D(a,b)>0$ and $f_{xx}(a,b)<0$, then $f(x,y)$ has a local maximum at $(a,b)$
\item
if $D(a,b)<0$, then $f(x,y)$ has a saddle point at $(a,b)$
\item
if $D(a,b)=0$, then we cannot draw any conclusions without more work
\end{itemize}
\end{theorem}

\end{alignat*}
\vspace{-1.5cm}
\end{theorem}

\begin{defn}[Dot Product]\label{def:dotProd}
The dot product of the vectors $\va$ and $\vb$ is denoted $\va\cdot\vb$
and is defined by
\begin{alignat*}{5}
&\va=\llt a_1,a_2,a_3\rgt ,\quad &&\vb=\llt b_1,b_2,b_3\rgt\quad &\implies\quad
&\va\cdot\vb = a_1b_1+a_2b_2+a_3b_3
\end{alignat*}
\end{defn}
\begin{theorem}[Properties of the Dot Product]\label{thm:dotPppties}
Let $\va$, $\vb$ and $\vc$ be vectors and let $s$ be a scalar. Then
\begin{alignat*}{3}
&(1) &&\;\va\cdot\va=\|\va\|^2\\
&(2) &&\;\va\cdot\vb=\vb\cdot\va\\
&(3) &&\;\va\cdot(\vb+\vc)=\va\cdot\vb+\va\cdot\vc,\quad (\va+\vb)\cdot\vc=\va\cdot\vc+\vb\cdot\vc\quad(distributive)\\
&(4) &&\;(s\va)\cdot\vb= s(\va\cdot\vb)\\
&(5) &&\;\vZero\cdot\va=0\\
&(6) &&\;\va\cdot\vb=\|\va\|\,\|\vb\|\,\cos\theta
\text{ where $0\leqslant\theta\leqslant\pi$ is the angle between $\va$ and $\vb$}\\
&(7) &&\;\va\cdot\vb=0\iff\va=\vZero\text{ or }\vb=\vZero
\text{ or } \va\perp\vb
\end{alignat*}
\end{theorem}
\begin{proof}
proof of (6):
\begin{efig}
\begin{center}
\includegraphics{cosineB.pdf}
\end{center}
\end{efig}
\begin{align*}
\|\va-\vb\|^2&=\big(\|\vb\|-\|\va\|\cos\theta\big)^2+\big(\|\va\|\sin\theta\big)^2 \\
&=\|\vb\|^2 - 2\|\va\|\,\|\vb\|\,\cos\theta + \|\va\|^2\cos^2\theta + \|\va\|^2\sin^2\theta \\
&=\|\vb\|^2 - 2\|\va\|\,\|\vb\|\,\cos\theta + \|\va\|^2
\end{align*}
\begin{align*}
\|\va-\vb\|^2 &= (\va - \vb)\cdot(\va - \vb) = \va\cdot\va - \va\cdot\vb - \vb\cdot\va - \vb\cdot\vb \\
&= \|\va\|^2 + \|\vb\|^2 - 2\va\cdot\vb = \|\vb\|^2 - 2\|\va\|\,\|\vb\|\,\cos\theta + \|\va\|^2
\end{align*}
So $\va\cdot\vb=|\va|\,|\vb|\,\cos\theta$.
\end{proof}

Compute the distance between $(2, 3, -1)$ and $\ds\frac{x-1}{1} = \frac{y-2}{1} = \frac{z-3}{2}$.
\soln

\item Let $\ell$ be the distance, then $\ell^2 = (t + 1 - 2)^2 + (t + 2 - 3)^2 + (2 t + 3 + 1)^2 = 6t^2 + 12 t + 18$
\item $(\ell^2)' = 12 t + 12\;\Longrightarrow\;t = -1$, so the minimum distance is $\sqrt{6 - 12 + 18} = 2\sqrt{3}$.
\end{itemize}

\begin{itemize}

\begin{theorem}[Arithmetic of differentiation]\label{thm:DIFFalgebra}
Let
\begin{itemize}\itemsep1pt \parskip0pt \parsep0pt %\itemindent-15pt
\item[$\circ$]
$\va(t),\vb(t)$ be vector valued differentiable functions of $t\in\bbbr$
with values in $\bbbr^n$
\item[$\circ$]
$\alpha,\beta \in \mathbb{R}$ be constants
\item[$\circ$]
$\ga(t)$ and $s(t)$ be real valued differentiable functions of $t\in\bbbr$
\end{itemize}
Then
\begin{alignat*}{5}
&\text{(a)}\quad &&\diff{}{t}\big(\alpha\,\va(t)+\beta\,\vb(t)\big)
=\alpha\,\va'(t)+\beta\,\vb'(t)
&&\text{(linear combination)}
\\
&\text{(b)} &&\diff{}{t}\big(\ga(t)\vb(t)\big)
=\ga'(t)\vb(t)+\ga(t)\vb'(t)
&&\text{(multiplication by scalar function)}
\\
&\text{(c)} &&\diff{}{t}\big(\va(t)\cdot\vb(t)\big)
=\va'(t)\cdot\vb(t)+\va(t)\cdot\vb'(t)
&&\text{(dot product)}
\\
&\text{(d)} &&\diff{}{t}\big(\va(t)\times\vb(t)\big)
=\va'(t)\times\vb(t)+\va(t)\times\vb'(t)
\ \ &&\text{(cross product)}
\\
&\text{(e)} &&\diff{}{t}\big(\va\big(s(t)\big)\big)
=\va'\big(s(t)\big)\,s'(t)
&&\text{(composition)}
\end{alignat*}
\end{theorem}
\begin{lemma}\label{lem:CVtanArclen}
Let $\vr(t)$ be a parametrized curve.
\begin{enumerate}[(a)]
\item
Denote by $\hat\vT(t)$ the unit tangent vector to the curve
at $\vr(t)$ pointing in the direction of increasing $t$. If
$\vr'(t)\ne 0$
then
\begin{equation*}
\hat\vT(t) = \frac{\vr'(t)}{|\vr'(t)|}
\end{equation*}
\item
Denote by $s(t)$ the length of the part of the curve between
$\vr(0)$ and $\vr(t)$. Then
\begin{align*}
\diff{s}{t}(t)&=\left|\diff{\vr}{t}(t)\right| \\
s(T)-s(T_0)&= \int_{T_0}^T \left|\diff{\vr}{t}(t)\right|\,\dee{t}\qquad
\raisebox{-20pt}{\smash{\includegraphics{parCurveDerivB.pdf}}}
\end{align*}
\item
In particular, if the parameter happens to be arc length, i.e. if $t=s$,
so that $\diff{s}{s}=1$, then
\begin{equation*}
\left|\diff{\vr}{s}(s)\right|=1\qquad
\hat\vT(s) = \vr'(s)
\end{equation*}
\end{enumerate}
\end{lemma}
\begin{lemma}\label{lem:CVvelocityEtc}
Given the position vector $\vr(t)=\llt x(t)\,,\,y(t)\,,\,z(t)\rgt$, then
\begin{align*}
\text{velocity at time $t$} &= \vv(t) = \vr'(t) = x'(t)\hi + y'(t)\hj + z'(t)\hk = \diff{s}{t}(t)\,\hat\vT(t)\\
\text{speed at time $t$} &= \diff{s}{t}(t) = \|\vv(t)\| = \|\vr'(t)\| = \sqrt{(x'(t)^2+y'(t)^2+z'(t)^2} \\
\text{acceleration at time $t$} &= \va(t) = \vr''(t) = \vv'(t) = x''(t)\hi + y''(t)\hj + z''(t)\hk \\
\intertext{the distance travelled between times $T_0$ and $T$}
s(T)-s(T_0) &= \int_{T_0}^T \Big\|\diff{\vr}{t}(t)\Big\|\,\dee{t} = \int_{T_0}^T \sqrt{(x'(t)^2+y'(t)^2+z'(t)^2}\,\dee{t}
\end{align*}
\end{lemma}
